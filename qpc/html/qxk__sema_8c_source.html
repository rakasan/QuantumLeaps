114 bool QXSemaphore_wait(QXSemaphore * const me, uint_fast16_t const nTicks) {
115  bool signaled = true; /* assume that the semaphore will be signaled */
116  QXThread *curr;
117  QF_CRIT_STAT_
118 
119  QF_CRIT_E_();
120  curr = QXK_PTR_CAST_(QXThread*, QXK_attr_.curr); /* volatile into temp. */
121 
128  Q_REQUIRE_ID(200, (!QXK_ISR_CONTEXT_()) /* can't wait inside an ISR */
129  && (me->max_count > 0U) /* sema must be initialized */
130  && (curr != (QXThread *)0) /* curr must be extended */
131  && (curr->super.super.temp.obj == (QMState *)0)); /* NOT blocked */
133  Q_REQUIRE_ID(201, QXK_attr_.lockHolder != curr->super.prio);
134 
135  if (me->count > 0U) {
136  --me->count; /* semaphore taken: decrement the count */
137  }
138  else {
139  uint_fast8_t p = (uint_fast8_t)curr->super.dynPrio;
140 
141  /* remember the blocking object (this semaphore) */
142  curr->super.super.temp.obj = QXK_PTR_CAST_(QMState*, me);
143  QXThread_teArm_(curr, (QSignal)QXK_SEMA_SIG, nTicks);
144 
145  /* remove this curr prio from the ready set (will block)
146  * and insert to the waiting set on this semaphore */
147  QPSet_insert(&me->waitSet, p); /* add to waiting-set */
148  QPSet_remove(&QXK_attr_.readySet, p); /* remove from ready-set */
149 
150  /* schedule the next thread if multitasking started */
151  (void)QXK_sched_();
152  QF_CRIT_X_();
153  QF_CRIT_EXIT_NOP(); /* BLOCK here !!! */
154 
155  QF_CRIT_E_(); /* AFTER unblocking... */
156  /* the blocking object must be this semaphore */
157  Q_ASSERT_ID(240, curr->super.super.temp.obj
158  == QXK_PTR_CAST_(QMState*, me));
159 
160  /* did the blocking time-out? (signal of zero means that it did) */
161  if (curr->timeEvt.super.sig == 0U) {
162  if (QPSet_hasElement(&me->waitSet, p)) { /* still waiting? */
163  QPSet_remove(&me->waitSet, p); /* remove unblocked thread */
164  signaled = false; /* the semaphore was NOT signaled */
165  /* semaphore NOT taken: do NOT decrement the count */
166  }
167  else { /* semaphore was both signaled and timed out */
168  --me->count; /* semaphore signaled: decrement the count */
169  }
170  }
171  else { /* blocking did NOT time out */
172  /* the thread must NOT be waiting on this semaphore */
173  Q_ASSERT_ID(250,!QPSet_hasElement(&me->waitSet, p));
174 
175  --me->count; /* semaphore signaled: decrement the count */
176  }
177  curr->super.super.temp.obj = (QMState *)0; /* clear blocking obj. */
178  }
179  QF_CRIT_X_();
180 
181  return signaled;
182 }

200  bool isAvailable;
201  QF_CRIT_STAT_
202 
204  Q_REQUIRE_ID(300, me->max_count > 0U);
205 
206  QF_CRIT_E_();
207  /* is the semaphore available? */

240 bool QXSemaphore_signal(QXSemaphore * const me) {
241  bool signaled = true; /* assume that the semaphore will be signaled */
242  QF_CRIT_STAT_
243 
245  Q_REQUIRE_ID(400, me->max_count > 0U);
246 
247  QF_CRIT_E_();
248  if (me->count < me->max_count) {
249 
250  ++me->count; /* increment the semaphore count */
251 
252  if (QPSet_notEmpty(&me->waitSet)) {
253  uint_fast8_t p;
254  QXThread *thr;
255 
256  /* find the highest-priority thread waiting on this semaphore */
257  QPSet_findMax(&me->waitSet, p);
258  thr = QXK_PTR_CAST_(QXThread*, QF_active_[p]);
259 
260  /* assert that the tread:
261  * - must be registered in QF;
262  * - must be extended; and
263  * - must be blocked on this semaphore;
264  */
265  Q_ASSERT_ID(410, (thr != (QXThread *)0)
266  && (thr->super.osObject != (struct QActive *)0)
267  && (thr->super.super.temp.obj
268  == QXK_PTR_CAST_(QMState*, me)));
269 
270  /* disarm the internal time event */
271  (void)QXThread_teDisarm_(thr);
272 
273  /* make the thread ready to run and remove from the wait-list */
274  QPSet_insert(&QXK_attr_.readySet, p);
275  QPSet_remove(&me->waitSet, p);
276 
277  if (!QXK_ISR_CONTEXT_()) { /* not inside ISR? */
278  (void)QXK_sched_(); /* schedule the next thread */
279  }
280  }
281  }
282  else {
283  signaled = false; /* semaphore NOT signaled */
284  }
285  QF_CRIT_X_();
286 
287  return signaled;
288 }
