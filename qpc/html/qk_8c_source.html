230 QSchedStatus QK_schedLock(uint_fast8_t ceiling) {
231  QSchedStatus stat;
232  QF_CRIT_STAT_
233  QF_CRIT_E_();
234 
238  Q_REQUIRE_ID(600, !QK_ISR_CONTEXT_());
239 
240  /* first store the previous lock prio */
241  if (QK_attr_.lockPrio < ceiling) { /* raising lock prio? */
242  stat = ((QSchedStatus)QK_attr_.lockPrio << 8);
243  QK_attr_.lockPrio = (uint8_t)ceiling;
244 
245  QS_BEGIN_NOCRIT_PRE_(QS_SCHED_LOCK, 0U)
246  QS_TIME_PRE_(); /* timestamp */
247  QS_2U8_PRE_(stat, /* the previous lock prio */
248  QK_attr_.lockPrio); /* the new lock prio */
249  QS_END_NOCRIT_PRE_()
250 
251  /* add the previous lock holder priority */
252  stat |= (QSchedStatus)QK_attr_.lockHolder;
253 
254  QK_attr_.lockHolder = QK_attr_.actPrio;
255  }
256  else {
257  stat = 0xFFU;
258  }
259  QF_CRIT_X_();
260 
261  return stat; /* return the status to be saved in a stack variable */
262 }

280 void QK_schedUnlock(QSchedStatus stat) {
281  /* has the scheduler been actually locked by the last QK_schedLock()? */
282  if (stat != 0xFFU) {
283  uint_fast8_t lockPrio = (uint_fast8_t)QK_attr_.lockPrio;
284  uint_fast8_t prevPrio = (uint_fast8_t)(stat >> 8);
285  QF_CRIT_STAT_
286  QF_CRIT_E_();
287 
292  Q_REQUIRE_ID(700, (!QK_ISR_CONTEXT_())
293  && (lockPrio > prevPrio));
294 
295  QS_BEGIN_NOCRIT_PRE_(QS_SCHED_UNLOCK, 0U)
296  QS_TIME_PRE_(); /* timestamp */
297  QS_2U8_PRE_(lockPrio, /* lock prio before unlocking */
298  prevPrio); /* lock prio after unlocking */
299  QS_END_NOCRIT_PRE_()
300 
301  /* restore the previous lock priority and lock holder */
302  QK_attr_.lockPrio = (uint8_t)prevPrio;
303  QK_attr_.lockHolder = (uint8_t)(stat & 0xFFU);
304 
305  /* find the highest-prio thread ready to run */
306  if (QK_sched_() != 0U) { /* priority found? */
307  QK_activate_(); /* activate any unlocked basic threads */
308  }
309 
310  QF_CRIT_X_();
311  }

329 uint_fast8_t QK_sched_(void) {
330  uint_fast8_t p; /* for priority */
331 
332  /* find the highest-prio AO with non-empty event queue */
333  QPSet_findMax(&QK_attr_.readySet, p);
334 
335  /* is the highest-prio below the active priority? */
336  if (p <= (uint_fast8_t)QK_attr_.actPrio) {
337  p = 0U; /* no activation needed */
338  }
339  else if (p <= (uint_fast8_t)QK_attr_.lockPrio) { /* below the lock prio?*/
340  p = 0U; /* no activation needed */
341  }
342  else {
343  Q_ASSERT_ID(410, p <= QF_MAX_ACTIVE);
344  QK_attr_.nextPrio = (uint8_t)p; /* next AO to run */
345  }
346  return p;
347 }

359 void QK_activate_(void) {
360  uint_fast8_t const pin = (uint_fast8_t)QK_attr_.actPrio; /* save */
361  uint_fast8_t p = (uint_fast8_t)QK_attr_.nextPrio; /* next prio to run */
362  QActive *a;
363 #if (defined QK_ON_CONTEXT_SW) || (defined Q_SPY)
364  uint_fast8_t pprev;
365 #endif /* QK_ON_CONTEXT_SW || Q_SPY */
366 
367  /* QK_attr_.actPrio and QK_attr_.nextPrio must be in ragne */
368  Q_REQUIRE_ID(500, (pin < Q_DIM(QF_active_))
369  && (0U < p)
370  && (p < Q_DIM(QF_active_)));
371 
372 #if (defined QK_ON_CONTEXT_SW) || (defined Q_SPY)
373  pprev = pin;
374 #endif /* QK_ON_CONTEXT_SW || Q_SPY */
375 
376  QK_attr_.nextPrio = 0U; /* clear for the next time */
377 
378  /* loop until no more ready-to-run AOs of higher prio than the initial */
379  do {
380  QEvt const *e;
381  a = QF_active_[p]; /* obtain the pointer to the AO */
382  QK_attr_.actPrio = (uint8_t)p; /* this becomes the active prio */
383 
384  QS_BEGIN_NOCRIT_PRE_(QS_SCHED_NEXT, a->prio)
385  QS_TIME_PRE_(); /* timestamp */
386  QS_2U8_PRE_(p, /* priority of the scheduled AO */
387  pprev); /* previous priority */
388  QS_END_NOCRIT_PRE_()
389 
390 #if (defined QK_ON_CONTEXT_SW) || (defined Q_SPY)
391  if (p != pprev) { /* changing threads? */
392 
393 #ifdef QK_ON_CONTEXT_SW
394  /* context-switch callback*/
395  QK_onContextSw(((pprev != 0U)
396  ? QF_active_[pprev]
397  : (QActive *)0), a);
398 #endif /* QK_ON_CONTEXT_SW */
399 
400  pprev = p; /* update previous priority */
401  }
402 #endif /* QK_ON_CONTEXT_SW || Q_SPY */
403 
404  QF_INT_ENABLE(); /* unconditionally enable interrupts */
405 
406  /* perform the run-to-completion (RTC) step...
407  * 1. retrieve the event from the AO's event queue, which by this
408  * time must be non-empty and QActive_get_() asserts it.
409  * 2. dispatch the event to the AO's state machine.
410  * 3. determine if event is garbage and collect it if so
411  */
412  e = QActive_get_(a);
413  QHSM_DISPATCH(&a->super, e, a->prio);
414  QF_gc(e);
415 
416  /* determine the next highest-priority AO ready to run... */
417  QF_INT_DISABLE(); /* unconditionally disable interrupts */
418 
419  if (a->eQueue.frontEvt == (QEvt *)0) { /* empty queue? */
420  QPSet_remove(&QK_attr_.readySet, p);
421  }
422 
423  /* find new highest-prio AO ready to run... */
424  QPSet_findMax(&QK_attr_.readySet, p);
425 
426  /* is the new priority below the initial preemption threshold? */
427  if (p <= pin) {
428  p = 0U;
429  }
430  else if (p <= (uint_fast8_t)QK_attr_.lockPrio) {/* below lock prio? */
431  p = 0U; /* active object not eligible */
432  }
433  else {
434  Q_ASSERT_ID(510, p <= QF_MAX_ACTIVE);
435  }
436  } while (p != 0U);
437 
438  QK_attr_.actPrio = (uint8_t)pin; /* restore the active priority */
439 
440 #if (defined QK_ON_CONTEXT_SW) || (defined Q_SPY)
441  if (pin != 0U) { /* resuming an active object? */
442  a = QF_active_[pin]; /* the pointer to the preempted AO */
443 
444  QS_BEGIN_NOCRIT_PRE_(QS_SCHED_RESUME, a->prio)
445  QS_TIME_PRE_(); /* timestamp */
446  QS_2U8_PRE_(pin, /* priority of the resumed AO */
447  pprev); /* previous priority */
448  QS_END_NOCRIT_PRE_()
449  }
450  else { /* resuming priority==0 --> idle */
451  a = (QActive *)0; /* QK idle loop */
452 
453  QS_BEGIN_NOCRIT_PRE_(QS_SCHED_IDLE, 0U)
454  QS_TIME_PRE_(); /* timestamp */
455  QS_U8_PRE_(pprev); /* previous priority */
456  QS_END_NOCRIT_PRE_()
457  }
458 
459 #ifdef QK_ON_CONTEXT_SW
460  QK_onContextSw(QF_active_[pprev], a); /* context-switch callback */
461 #endif /* QK_ON_CONTEXT_SW */
462 
463 #endif /* QK_ON_CONTEXT_SW || Q_SPY */
464 }
