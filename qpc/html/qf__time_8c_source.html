78 void QF_tickX_(uint_fast8_t const tickRate, void const * const sender)
79 #else
80 void QF_tickX_(uint_fast8_t const tickRate)
81 #endif
82 {
83  QTimeEvt *prev = &QF_timeEvtHead_[tickRate];
84  QF_CRIT_STAT_
85 
86  QF_CRIT_E_();
87 
88  QS_BEGIN_NOCRIT_PRE_(QS_QF_TICK, 0U)
89  ++prev->ctr;
90  QS_TEC_PRE_(prev->ctr); /* tick ctr */
91  QS_U8_PRE_(tickRate); /* tick rate */
92  QS_END_NOCRIT_PRE_()
93 
94  /* scan the linked-list of time events at this rate... */
95  for (;;) {
96  QTimeEvt *t = prev->next; /* advance down the time evt. list */
97 
98  /* end of the list? */
99  if (t == (QTimeEvt *)0) {
100 
101  /* any new time events armed since the last run of QF_tickX_()? */
102  if (QF_timeEvtHead_[tickRate].act != (void *)0) {
103 
104  /* sanity check */
105  Q_ASSERT_CRIT_(110, prev != (QTimeEvt *)0);
106  prev->next = (QTimeEvt *)QF_timeEvtHead_[tickRate].act;
107  QF_timeEvtHead_[tickRate].act = (void *)0;
108  t = prev->next; /* switch to the new list */
109  }
110  else {
111  break; /* all currently armed time evts. processed */
112  }
113  }
114 
115  /* time event scheduled for removal? */
116  if (t->ctr == 0U) {
117  prev->next = t->next;
118  /* mark time event 't' as NOT linked */
119  t->super.refCtr_ &= (uint8_t)(~TE_IS_LINKED & 0xFFU);
120  /* do NOT advance the prev pointer */
121  QF_CRIT_X_(); /* exit crit. section to reduce latency */
122 
123  /* prevent merging critical sections, see NOTE1 below */
124  QF_CRIT_EXIT_NOP();
125  }
126  else {
127  --t->ctr;
128 
129  /* is time event about to expire? */
130  if (t->ctr == 0U) {
131  QActive *act = (QActive *)t->act; /* temp. for volatile */
132 
133  /* periodic time evt? */
134  if (t->interval != 0U) {
135  t->ctr = t->interval; /* rearm the time event */
136  prev = t; /* advance to this time event */
137  }
138  /* one-shot time event: automatically disarm */
139  else {
140  prev->next = t->next;
141  /* mark time event 't' as NOT linked */
142  t->super.refCtr_ &= (uint8_t)(~TE_IS_LINKED & 0xFFU);
143  /* do NOT advance the prev pointer */
144 
145  QS_BEGIN_NOCRIT_PRE_(QS_QF_TIMEEVT_AUTO_DISARM, act->prio)
146  QS_OBJ_PRE_(t); /* this time event object */
147  QS_OBJ_PRE_(act); /* the target AO */
148  QS_U8_PRE_(tickRate); /* tick rate */
149  QS_END_NOCRIT_PRE_()
150  }
151 
152  QS_BEGIN_NOCRIT_PRE_(QS_QF_TIMEEVT_POST, act->prio)
153  QS_TIME_PRE_(); /* timestamp */
154  QS_OBJ_PRE_(t); /* the time event object */
155  QS_SIG_PRE_(t->super.sig); /* signal of this time event */
156  QS_OBJ_PRE_(act); /* the target AO */
157  QS_U8_PRE_(tickRate); /* tick rate */
158  QS_END_NOCRIT_PRE_()
159 
160  QF_CRIT_X_(); /* exit critical section before posting */
161 
162  /* QACTIVE_POST() asserts internally if the queue overflows */
163  QACTIVE_POST(act, &t->super, sender);
164  }
165  else {
166  prev = t; /* advance to this time event */
167  QF_CRIT_X_(); /* exit crit. section to reduce latency */
168 
169  /* prevent merging critical sections, see NOTE1 below */
170  QF_CRIT_EXIT_NOP();
171  }
172  }
173  QF_CRIT_E_(); /* re-enter crit. section to continue */
174  }
175  QF_CRIT_X_();
176 }

178 /*****************************************************************************
179 * NOTE1:
180 * In some QF ports the critical section exit takes effect only on the next
181 * machine instruction. If this case, the next instruction is another entry
182 * to a critical section, the critical section won't be really exited, but
183 * rather the two adjacent critical sections would be merged.
184 *
185 * The QF_CRIT_EXIT_NOP() macro contains minimal code required
186 * to prevent such merging of critical sections in QF ports,
187 * in which it can occur.

238 void QTimeEvt_ctorX(QTimeEvt * const me, QActive * const act,
239  enum_t const sig, uint_fast8_t tickRate)
240 {
242  Q_REQUIRE_ID(300, (sig >= (enum_t)Q_USER_SIG)
243  && (tickRate < QF_MAX_TICK_RATE));
244 
245  me->next = (QTimeEvt *)0;
246  me->ctr = 0U;
247  me->interval = 0U;
248  me->super.sig = (QSignal)sig;
249 
250  /* For backwards compatibility with QTimeEvt_ctor(), the active object
251  * pointer can be uninitialized (NULL) and is NOT validated in the
252  * precondition. The active object pointer is validated in preconditions
253  * to QTimeEvt_arm_() and QTimeEvt_rearm().
254  */
255  me->act = act;
256 
257  /* Setting the POOL_ID event attribute to zero is correct only for
258  * events not allocated from event pools, which must be the case
259  * for Time Events.
260  */
261  me->super.poolId_ = 0U;
262 
263  /* The refCtr_ attribute is not used in time events, so it is
264  * reused to hold the tickRate as well as other information
265  */
266  me->super.refCtr_ = (uint8_t)tickRate;
267 }

297 void QTimeEvt_armX(QTimeEvt * const me,
298  QTimeEvtCtr const nTicks, QTimeEvtCtr const interval)
299 {
300  uint_fast8_t tickRate = ((uint_fast8_t)me->super.refCtr_ & TE_TICK_RATE);
301  QTimeEvtCtr ctr = me->ctr;
302 #ifdef Q_SPY
303  uint_fast8_t const qs_id = ((QActive *)(me->act))->prio;
304 #endif
305  QF_CRIT_STAT_
306 
310  Q_REQUIRE_ID(400, (me->act != (void *)0)
311  && (ctr == 0U)
312  && (nTicks != 0U)
313  && (tickRate < (uint_fast8_t)QF_MAX_TICK_RATE)
314  && (me->super.sig >= (QSignal)Q_USER_SIG));
315 #ifdef Q_NASSERT
316  (void)ctr; /* avoid compiler warning about unused variable */
317 #endif
318 
319  QF_CRIT_E_();
320  me->ctr = nTicks;
321  me->interval = interval;
322 
323  /* is the time event unlinked?
324  * NOTE: For the duration of a single clock tick of the specified tick
325  * rate a time event can be disarmed and yet still linked into the list,
326  * because un-linking is performed exclusively in the QF_tickX() function.
327  */
328  if ((me->super.refCtr_ & TE_IS_LINKED) == 0U) {
329  me->super.refCtr_ |= TE_IS_LINKED; /* mark as linked */
330 
331  /* The time event is initially inserted into the separate
332  * "freshly armed" link list based on QF_timeEvtHead_[tickRate].act.
333  * Only later, inside the QF_tickX() function, the "freshly armed"
334  * list is appended to the main list of armed time events based on
335  * QF_timeEvtHead_[tickRate].next. Again, this is to keep any
336  * changes to the main list exclusively inside the QF_tickX()
337  * function.
338  */
339  me->next = (QTimeEvt *)QF_timeEvtHead_[tickRate].act;
340  QF_timeEvtHead_[tickRate].act = me;
341  }
342 
343  QS_BEGIN_NOCRIT_PRE_(QS_QF_TIMEEVT_ARM, qs_id)
344  QS_TIME_PRE_(); /* timestamp */
345  QS_OBJ_PRE_(me); /* this time event object */
346  QS_OBJ_PRE_(me->act); /* the active object */
347  QS_TEC_PRE_(nTicks); /* the number of ticks */
348  QS_TEC_PRE_(interval); /* the interval */
349  QS_U8_PRE_(tickRate); /* tick rate */
350  QS_END_NOCRIT_PRE_()
351 
352  QF_CRIT_X_();
353 }

374 bool QTimeEvt_disarm(QTimeEvt * const me) {
375  bool wasArmed;
376 #ifdef Q_SPY
377  uint_fast8_t const qs_id = ((QActive *)(me->act))->prio;
378 #endif
379  QF_CRIT_STAT_
380 
381  QF_CRIT_E_();
382 
383  /* is the time event actually armed? */
384  if (me->ctr != 0U) {
385  wasArmed = true;
386  me->super.refCtr_ |= TE_WAS_DISARMED;
387 
388  QS_BEGIN_NOCRIT_PRE_(QS_QF_TIMEEVT_DISARM, qs_id)
389  QS_TIME_PRE_(); /* timestamp */
390  QS_OBJ_PRE_(me); /* this time event object */
391  QS_OBJ_PRE_(me->act); /* the target AO */
392  QS_TEC_PRE_(me->ctr); /* the number of ticks */
393  QS_TEC_PRE_(me->interval); /* the interval */
394  QS_U8_PRE_(me->super.refCtr_ & TE_TICK_RATE);
395  QS_END_NOCRIT_PRE_()
396 
397  me->ctr = 0U; /* schedule removal from the list */
398  }
399  else { /* the time event was already disarmed automatically */
400  wasArmed = false;
401  me->super.refCtr_ &= (uint8_t)(~TE_WAS_DISARMED & 0xFFU);
402 
403  QS_BEGIN_NOCRIT_PRE_(QS_QF_TIMEEVT_DISARM_ATTEMPT, qs_id)
404  QS_TIME_PRE_(); /* timestamp */
405  QS_OBJ_PRE_(me); /* this time event object */
406  QS_OBJ_PRE_(me->act); /* the target AO */
407  QS_U8_PRE_(me->super.refCtr_ & TE_TICK_RATE);
408  QS_END_NOCRIT_PRE_()
409 
410  }
411  QF_CRIT_X_();
412  return wasArmed;
413 }

436 bool QTimeEvt_rearm(QTimeEvt * const me, QTimeEvtCtr const nTicks) {
437  uint_fast8_t tickRate = (uint_fast8_t)me->super.refCtr_ & TE_TICK_RATE;
438  bool wasArmed;
439 #ifdef Q_SPY
440  uint_fast8_t const qs_id = ((QActive *)(me->act))->prio;
441 #endif
442  QF_CRIT_STAT_
443 
447  Q_REQUIRE_ID(600, (me->act != (void *)0)
448  && (tickRate < QF_MAX_TICK_RATE)
449  && (nTicks != 0U)
450  && (me->super.sig >= (QSignal)Q_USER_SIG));
451 
452  QF_CRIT_E_();
453 
454  /* is the time evt not running? */
455  if (me->ctr == 0U) {
456  wasArmed = false;
457 
458  /* NOTE: For the duration of a single clock tick of the specified
459  * tick rate a time event can be disarmed and yet still linked into
460  * the list, because unlinking is performed exclusively in the
461  * QF_tickX() function.
462  */
463  /* is the time event linked yet? */
464  if ((me->super.refCtr_ & TE_IS_LINKED) == 0U) {
465  me->super.refCtr_ |= TE_IS_LINKED; /* mark as linked */
466 
467  /* The time event is initially inserted into the separate
468  * "freshly armed" list based on QF_timeEvtHead_[tickRate].act.
469  * Only later, inside the QF_tickX() function, the "freshly armed"
470  * list is appended to the main list of armed time events based on
471  * QF_timeEvtHead_[tickRate].next. Again, this is to keep any
472  * changes to the main list exclusively inside the QF_tickX()
473  * function.
474  */
475  me->next = (QTimeEvt *)QF_timeEvtHead_[tickRate].act;
476  QF_timeEvtHead_[tickRate].act = me;
477  }
478  }
479  else { /* the time event was armed */
480  wasArmed = true;
481  }
482  me->ctr = nTicks; /* re-load the tick counter (shift the phasing) */
483 
484  QS_BEGIN_NOCRIT_PRE_(QS_QF_TIMEEVT_REARM, qs_id)
485  QS_TIME_PRE_(); /* timestamp */
486  QS_OBJ_PRE_(me); /* this time event object */
487  QS_OBJ_PRE_(me->act); /* the target AO */
488  QS_TEC_PRE_(me->ctr); /* the number of ticks */
489  QS_TEC_PRE_(me->interval); /* the interval */
490  QS_2U8_PRE_(tickRate, (wasArmed ? 1U : 0U));
491  QS_END_NOCRIT_PRE_()
492 
493  QF_CRIT_X_();
494  return wasArmed;
495 }
