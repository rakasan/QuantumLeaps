252  if (r >= (QState)Q_RET_HANDLED) {

277  }

282  if (r >= (QState)Q_RET_TRAN) {
283 #ifdef Q_SPY
284  QMState const *ts = t; /* transition source for QS tracing */

324  me->state.obj = s; /* restore the original state */
325  r = (*tmp.act)(me); /* execute the XP action */
326  if (r == (QState)Q_RET_TRAN) { /* XP -> TRAN ? */

341  }

343  tmp.obj = me->state.obj; /* save the history */
344  me->state.obj = s; /* restore the original state */
345 #ifdef Q_SPY
