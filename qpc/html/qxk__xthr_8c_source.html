289  QF_CRIT_STAT_
290  QS_TEST_PROBE_DEF(&QXThread_post_)
291 
292  /* is it the private time event? */
293  if (e == &QXTHREAD_CAST_(me)->timeEvt.super) {
294  QF_CRIT_E_();
295  /* the private time event is disarmed and not in any queue,
296  * so it is safe to change its signal. The signal of 0 means
297  * that the time event has expired.
298  */
299  QXTHREAD_CAST_(me)->timeEvt.super.sig = 0U;
300 
301  QXThread_unblock_(QXTHREAD_CAST_(me));
302  QF_CRIT_X_();
303 
304  status = true;
305  }
306  /* is the event queue provided? */
307  else if (me->eQueue.end != 0U) {
308  QEQueueCtr nFree; /* temporary to avoid UB for volatile access */
309 
311  Q_REQUIRE_ID(300, e != (QEvt *)0);
312 
313  QF_CRIT_E_();
314  nFree = me->eQueue.nFree; /* get volatile into the temporary */
315 
316  /* test-probe#1 for faking queue overflow */
317  QS_TEST_PROBE_ID(1,
318  nFree = 0U;
319  )
320 
321  if (margin == QF_NO_MARGIN) {
322  if (nFree > 0U) {
323  status = true; /* can post */
324  }
325  else {
326  status = false; /* cannot post */
327  Q_ERROR_CRIT_(310); /* must be able to post the event */
328  }
329  }
330  else if (nFree > (QEQueueCtr)margin) {
331  status = true; /* can post */
332  }
333  else {
334  status = false; /* cannot post, but don't assert */
335  }
336

338  if (e->poolId_ != 0U) {
339  QF_EVT_REF_CTR_INC_(e); /* increment the reference counter */
340  }
341 
342  if (status) { /* can post the event? */
343 
344  --nFree; /* one free entry just used up */
345  me->eQueue.nFree = nFree; /* update the volatile */
346  if (me->eQueue.nMin > nFree) {
347  me->eQueue.nMin = nFree; /* update minimum so far */
348  }

351  QS_TIME_PRE_(); /* timestamp */
352  QS_OBJ_PRE_(sender); /* the sender object */
353  QS_SIG_PRE_(e->sig); /* the signal of the event */
354  QS_OBJ_PRE_(me); /* this active object (recipient) */
355  QS_2U8_PRE_(e->poolId_, e->refCtr_); /* pool Id & ref Count */
356  QS_EQC_PRE_(nFree); /* number of free entries */
357  QS_EQC_PRE_(me->eQueue.nMin); /* min number of free entries */
358  QS_END_NOCRIT_PRE_()
359

361  if (me->eQueue.frontEvt == (QEvt *)0) {
362  me->eQueue.frontEvt = e; /* deliver event directly */
363 
364  /* is this thread blocked on the queue? */
365  if (me->super.temp.obj

367  {

371  if (!QXK_ISR_CONTEXT_()) {
372  (void)QXK_sched_();
373  }
374  }
375  }
376  /* queue is not empty, insert event into the ring-buffer */
377  else {
378  /* insert event into the ring buffer (FIFO) */
379  QF_PTR_AT_(me->eQueue.ring, me->eQueue.head) = e;
380 
381  /* need to wrap the head counter? */
382  if (me->eQueue.head == 0U) {
383  me->eQueue.head = me->eQueue.end; /* wrap around */
384  }
385  --me->eQueue.head; /* advance the head (counter clockwise) */
386  }
387 
388  QF_CRIT_X_();
389  }
390  else { /* cannot post the event */
391 
392  QS_BEGIN_NOCRIT_PRE_(QS_QF_ACTIVE_POST_ATTEMPT, me->prio)
393  QS_TIME_PRE_(); /* timestamp */
394  QS_OBJ_PRE_(sender); /* the sender object */
395  QS_SIG_PRE_(e->sig); /* the signal of the event */
396  QS_OBJ_PRE_(me); /* this active object (recipient) */
397  QS_2U8_PRE_(e->poolId_, e->refCtr_); /* pool Id & ref Count */
398  QS_EQC_PRE_(nFree); /* number of free entries */
399  QS_EQC_PRE_(margin); /* margin requested */
400  QS_END_NOCRIT_PRE_()
401 
402  QF_CRIT_X_();
403 
404  QF_gc(e); /* recycle the event to avoid a leak */
405  }
406  }
407  else { /* the queue is not available */
408  QF_gc(e); /* make sure the event is not leaked */
409  status = false;
410  Q_ERROR_ID(320); /* this extended thread cannot accept events */
411  }
412

453 QEvt const *QXThread_queueGet(uint_fast16_t const nTicks) {
454  QXThread *thr;
455  QEQueueCtr nFree;
456  QEvt const *e;
457  QF_CRIT_STAT_
458 
459  QF_CRIT_E_();
460  thr = QXTHREAD_CAST_(QXK_attr_.curr);
461 
467  Q_REQUIRE_ID(500, (!QXK_ISR_CONTEXT_()) /* can't block inside an ISR */
468  && (thr != (QXThread *)0) /* current thread must be extended */
469  && (thr->super.super.temp.obj == (QMState *)0)); /* !blocked */
471  Q_REQUIRE_ID(501, QXK_attr_.lockHolder != thr->super.prio);
472 
473  /* is the queue empty? */
474  if (thr->super.eQueue.frontEvt == (QEvt *)0) {
475 
476  /* remember the blocking object (the thread's queue) */
477  thr->super.super.temp.obj
478  = QXK_PTR_CAST_(QMState const*, &thr->super.eQueue);
479 
480  QXThread_teArm_(thr, (QSignal)QXK_QUEUE_SIG, nTicks);
481  QPSet_remove(&QXK_attr_.readySet, (uint_fast8_t)thr->super.dynPrio);
482  (void)QXK_sched_();
483  QF_CRIT_X_();
484  QF_CRIT_EXIT_NOP(); /* BLOCK here */
485 
486  QF_CRIT_E_();
487  /* the blocking object must be this queue */
488  Q_ASSERT_ID(510, thr->super.super.temp.obj
489  == QXK_PTR_CAST_(QMState const*, &thr->super.eQueue));
490  thr->super.super.temp.obj = (QMState *)0; /* clear */
491  }
492 
493  /* is the queue not empty? */
494  if (thr->super.eQueue.frontEvt != (QEvt *)0) {
495  e = thr->super.eQueue.frontEvt; /* always remove from the front */
496  nFree= thr->super.eQueue.nFree + 1U; /* volatile into tmp */
497  thr->super.eQueue.nFree = nFree; /* update the number of free */
498 
499  /* any events in the ring buffer? */
500  if (nFree <= thr->super.eQueue.end) {
501 
502  /* remove event from the tail */
503  thr->super.eQueue.frontEvt =
504  QF_PTR_AT_(thr->super.eQueue.ring, thr->super.eQueue.tail);
505  if (thr->super.eQueue.tail == 0U) { /* need to wrap? */
506  thr->super.eQueue.tail = thr->super.eQueue.end; /* wrap */
507  }
508  --thr->super.eQueue.tail;
509 
510  QS_BEGIN_NOCRIT_PRE_(QS_QF_ACTIVE_GET, thr->super.prio)
511  QS_TIME_PRE_(); /* timestamp */
512  QS_SIG_PRE_(e->sig); /* the signal of this event */
513  QS_OBJ_PRE_(&thr->super); /* this active object */
514  QS_2U8_PRE_(e->poolId_, e->refCtr_); /* pool Id & ref Count */
515  QS_EQC_PRE_(nFree); /* number of free entries */
516  QS_END_NOCRIT_PRE_()
517  }
518  else {
519  thr->super.eQueue.frontEvt = (QEvt *)0; /* empty queue */
520 
521  /* all entries in the queue must be free (+1 for fronEvt) */
522  Q_ASSERT_ID(520, nFree == (thr->super.eQueue.end + 1U));
523 
524  QS_BEGIN_NOCRIT_PRE_(QS_QF_ACTIVE_GET_LAST, thr->super.prio)
525  QS_TIME_PRE_(); /* timestamp */
526  QS_SIG_PRE_(e->sig); /* the signal of this event */
527  QS_OBJ_PRE_(&thr->super); /* this active object */
528  QS_2U8_PRE_(e->poolId_, e->refCtr_); /* pool Id & ref Count */
529  QS_END_NOCRIT_PRE_()
530  }
531  }
532  else { /* the queue is still empty -- the timeout must have fired */
533  e = (QEvt *)0;
534  }
535  QF_CRIT_X_();
536 
537  return e;
538 }

582 void QXThread_teArm_(QXThread * const me, QSignal sig,
583  uint_fast16_t const nTicks)
584 {
586  Q_REQUIRE_ID(700, me->timeEvt.ctr == 0U);
587 
588  me->timeEvt.super.sig = sig;
589 
590  if (nTicks != QXTHREAD_NO_TIMEOUT) {
591  me->timeEvt.ctr = (QTimeEvtCtr)nTicks;
592  me->timeEvt.interval = 0U;
593 
594  /* is the time event unlinked?
595  * NOTE: For the duration of a single clock tick of the specified tick
596  * rate a time event can be disarmed and yet still linked in the list,
597  * because un-linking is performed exclusively in QF_tickX().
598  */
599  if ((me->timeEvt.super.refCtr_ & TE_IS_LINKED) == 0U) {
600  uint_fast8_t tickRate = ((uint_fast8_t)me->timeEvt.super.refCtr_
601  & TE_TICK_RATE);
602  Q_ASSERT_ID(710, tickRate < Q_DIM(QF_timeEvtHead_));
603 
604  me->timeEvt.super.refCtr_ |= TE_IS_LINKED;
605 
606  /* The time event is initially inserted into the separate
607  * "freshly armed" list based on QF_timeEvtHead_[tickRate].act.
608  * Only later, inside the QF_tickX() function, the "freshly armed"
609  * list is appended to the main list of armed time events based on
610  * QF_timeEvtHead_[tickRate].next. Again, this is to keep any
611  * changes to the main list exclusively inside QF_tickX().
612  */
613  me->timeEvt.next
614  = QXK_PTR_CAST_(QTimeEvt*, QF_timeEvtHead_[tickRate].act);
615  QF_timeEvtHead_[tickRate].act = &me->timeEvt;
616  }
617  }
