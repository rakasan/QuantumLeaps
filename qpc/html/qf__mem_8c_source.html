89 void QMPool_init(QMPool * const me, void * const poolSto,
90  uint_fast32_t poolSize, uint_fast16_t blockSize)
91 {
92  QFreeBlock *fb;
93  uint_fast16_t nblocks;
94 
99  Q_REQUIRE_ID(100, (poolSto != (void *)0)
100  && (poolSize >= sizeof(QFreeBlock))
101  && ((blockSize + sizeof(QFreeBlock)) > blockSize));
102 
103  me->free_head = poolSto;
104 
105  /* round up the blockSize to fit an integer # free blocks, no division */
106  me->blockSize = (QMPoolSize)sizeof(QFreeBlock); /* start with just one */
107  nblocks = 1U;/* #free blocks that fit in one memory block */
108  while (me->blockSize < (QMPoolSize)blockSize) {
109  me->blockSize += (QMPoolSize)sizeof(QFreeBlock);
110  ++nblocks;
111  }
112  blockSize = (uint_fast16_t)me->blockSize; /* round-up to nearest block */
113 
114  /* the pool buffer must fit at least one rounded-up block */
115  Q_ASSERT_ID(110, poolSize >= blockSize);
116 
117  /* chain all blocks together in a free-list... */
118  poolSize -= (uint_fast32_t)blockSize; /* don't count the last block */
119  me->nTot = 1U; /* the last block already in the pool */
120  fb = (QFreeBlock *)me->free_head; /* start at the head of the free list */
121 
122  /* chain all blocks together in a free-list... */
123  while (poolSize >= (uint_fast32_t)blockSize) {
124  fb->next = &QF_PTR_AT_(fb, nblocks);/*point next link to next block */
125  fb = fb->next; /* advance to the next block */
126  poolSize -= (uint_fast32_t)blockSize; /* reduce available pool size */
127  ++me->nTot; /* increment the number of blocks so far */
128  }
129 
130  fb->next = (QFreeBlock *)0; /* the last link points to NULL */
131  me->nFree = me->nTot; /* all blocks are free */
132  me->nMin = me->nTot; /* the minimum number of free blocks */
133  me->start = poolSto; /* the original start this pool buffer */
134  me->end = fb; /* the last block in this pool */
135 }

213 void *QMPool_get(QMPool * const me, uint_fast16_t const margin,
214  uint_fast8_t const qs_id)
215 {
216  QFreeBlock *fb;
217  QF_CRIT_STAT_
218 
219  (void)qs_id; /* unused parameter (outside Q_SPY build configuration) */
220 
221  QF_CRIT_E_();
222 
223  /* have more free blocks than the requested margin? */
224  if (me->nFree > (QMPoolCtr)margin) {
225  void *fb_next;
226  fb = (QFreeBlock *)me->free_head; /* get a free block */
227 
228  /* the pool has some free blocks, so a free block must be available */
229  Q_ASSERT_CRIT_(310, fb != (QFreeBlock *)0);
230 
231  fb_next = fb->next; /* put volatile to a temporary to avoid UB */
232 
233  /* is the pool becoming empty? */
234  --me->nFree; /* one less free block */
235  if (me->nFree == 0U) {
236  /* pool is becoming empty, so the next free block must be NULL */
237  Q_ASSERT_CRIT_(320, fb_next == (QFreeBlock *)0);
238 
239  me->nMin = 0U; /* remember that the pool got empty */
240  }
241  else {
242  /* pool is not empty, so the next free block must be in range
243  *
244  * NOTE: the next free block pointer can fall out of range
245  * when the client code writes past the memory block, thus
246  * corrupting the next block.
247  */
248  Q_ASSERT_CRIT_(330, QF_PTR_RANGE_(fb_next, me->start, me->end));
249 
250  /* is the number of free blocks the new minimum so far? */
251  if (me->nMin > me->nFree) {
252  me->nMin = me->nFree; /* remember the new minimum */
253  }
254  }
255 
256  me->free_head = fb_next; /* set the head to the next free block */
257 
258  QS_BEGIN_NOCRIT_PRE_(QS_QF_MPOOL_GET, qs_id)
259  QS_TIME_PRE_(); /* timestamp */
260  QS_OBJ_PRE_(me); /* this memory pool */
261  QS_MPC_PRE_(me->nFree); /* # of free blocks in the pool */
262  QS_MPC_PRE_(me->nMin); /* min # free blocks ever in the pool */
263  QS_END_NOCRIT_PRE_()
264  }
265  /* don't have enough free blocks at this point */
266  else {
267  fb = (QFreeBlock *)0;
268 
269  QS_BEGIN_NOCRIT_PRE_(QS_QF_MPOOL_GET_ATTEMPT, qs_id)
270  QS_TIME_PRE_(); /* timestamp */
271  QS_OBJ_PRE_(me); /* this memory pool */
272  QS_MPC_PRE_(me->nFree); /* the number of free blocks in the pool */
273  QS_MPC_PRE_(margin); /* the requested margin */
274  QS_END_NOCRIT_PRE_()
275  }
276  QF_CRIT_X_();
277 
278  return fb; /* return the block or NULL pointer to the caller */
279 }
