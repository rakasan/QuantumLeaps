207  QS_TEST_PROBE_ID(1,
208  status = false;
209  if (margin == QF_NO_MARGIN) {
210  /* fake assertion Mod=qf_actq,Loc=110 */
211  Q_onAssert("qf_actq", 110);

216

220  }

297  /* return immediately (do nothing) for Test Probe != 0 */
298  QS_TEST_PROBE(return;)

309  * 1. retrieve the event from the AO's event queue, which by this
310  * time must be non-empty and The "Vanialla" kernel asserts it.

324 /* The testing version of system tick processing performs as follows:
325 * 1. If the Current Time Event (TE) Object is defined and the TE is armed,
326 * the TE is disarmed (if one-shot) and then posted to the recipient AO.
327 * 2. The linked-list of all armed Time Events is updated.
328 */
329 void QS_tickX_(uint_fast8_t const tickRate, void const * const sender) {
330  QTimeEvt *t;
331  QActive *act;
332  QTimeEvt *prev;
333  QF_CRIT_STAT_
334 
335  QF_CRIT_E_();
336  prev = &QF_timeEvtHead_[tickRate];
337 
338  QS_BEGIN_NOCRIT_PRE_(QS_QF_TICK, 0U)
339  ++prev->ctr;
340  QS_TEC_PRE_(prev->ctr); /* tick ctr */
341  QS_U8_PRE_(tickRate); /* tick rate */
342  QS_END_NOCRIT_PRE_()
343 
344  /* is current Time Event object provided? */
345  t = (QTimeEvt *)QS_rxPriv_.currObj[TE_OBJ];
346  if (t != (QTimeEvt *)0) {
347 
348  /* the time event must be armed */
349  Q_ASSERT_ID(810, t->ctr != 0U);
350 
351  act = (QActive *)(t->act); /* temp. for volatile */
352 
353  /* the recipient AO must be provided */
354  Q_ASSERT_ID(820, act != (QActive *)0);
355 
356  /* periodic time evt? */
357  if (t->interval != 0U) {
358  t->ctr = t->interval; /* rearm the time event */
359  }
360  else { /* one-shot time event: automatically disarm */
361  t->ctr = 0U; /* auto-disarm */
362  /* mark time event 't' as NOT linked */
363  t->super.refCtr_ &= (uint8_t)(~(uint8_t)TE_IS_LINKED);
364 
365  QS_BEGIN_NOCRIT_PRE_(QS_QF_TIMEEVT_AUTO_DISARM, act->prio)
366  QS_OBJ_PRE_(t); /* this time event object */
367  QS_OBJ_PRE_(act); /* the target AO */
368  QS_U8_PRE_(tickRate); /* tick rate */
369  QS_END_NOCRIT_PRE_()
370  }
371 
372  QS_BEGIN_NOCRIT_PRE_(QS_QF_TIMEEVT_POST, act->prio)
373  QS_TIME_PRE_(); /* timestamp */
374  QS_OBJ_PRE_(t); /* the time event object */
375  QS_SIG_PRE_(t->super.sig); /* signal of this time event */
376  QS_OBJ_PRE_(act); /* the target AO */
377  QS_U8_PRE_(tickRate); /* tick rate */
378  QS_END_NOCRIT_PRE_()
379 
380  QF_CRIT_X_(); /* exit critical section before posting */
381 
382  QACTIVE_POST(act, &t->super, sender); /* asserts if queue overflows */
383 
384  QF_CRIT_E_();
385  }
386 
387  /* update the linked list of time events */
388  for (;;) {
389  t = prev->next; /* advance down the time evt. list */
390 
391  /* end of the list? */
392  if (t == (QTimeEvt *)0) {
393 
394  /* any new time events armed since the last run of QF_tickX_()? */
395  if (QF_timeEvtHead_[tickRate].act != (void *)0) {
396 
397  /* sanity check */
398  Q_ASSERT_CRIT_(830, prev != (QTimeEvt *)0);
399  prev->next = (QTimeEvt *)QF_timeEvtHead_[tickRate].act;
400  QF_timeEvtHead_[tickRate].act = (void *)0;
401  t = prev->next; /* switch to the new list */
402  }
403  else {
404  break; /* all currently armed time evts. processed */
405  }
406  }
407 
408  /* time event scheduled for removal? */
409  if (t->ctr == 0U) {
410  prev->next = t->next;
411  /* mark time event 't' as NOT linked */
412  t->super.refCtr_ &= (uint8_t)(~(uint8_t)TE_IS_LINKED);
413  /* do NOT advance the prev pointer */
414  QF_CRIT_X_(); /* exit crit. section to reduce latency */
415 
416  /* prevent merging critical sections, see NOTE1 below */
417  QF_CRIT_EXIT_NOP();
418  }
419  else {
420  prev = t; /* advance to this time event */
421  QF_CRIT_X_(); /* exit crit. section to reduce latency */
422 
423  /* prevent merging critical sections, see NOTE1 below */
424  QF_CRIT_EXIT_NOP();
425  }
426  QF_CRIT_E_(); /* re-enter crit. section to continue */
427  }
428 
429  QF_CRIT_X_();
430 }
