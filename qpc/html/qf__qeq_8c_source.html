115 bool QEQueue_post(QEQueue * const me, QEvt const * const e,
116  uint_fast16_t const margin, uint_fast8_t const qs_id)
117 {
118  QEQueueCtr nFree; /* temporary to avoid UB for volatile access */
119  bool status;
120  QF_CRIT_STAT_
121 
122  /* @pre event must be valid */
123  Q_REQUIRE_ID(200, e != (QEvt *)0);
124 
125  (void)qs_id; /* unused parameter (outside Q_SPY build configuration) */
126 
127  QF_CRIT_E_();
128  nFree = me->nFree; /* get volatile into the temporary */
129 
130  /* required margin available? */
131  if (((margin == QF_NO_MARGIN) && (nFree > 0U))
132  || (nFree > (QEQueueCtr)margin))
133  {
134  /* is it a dynamic event? */
135  if (e->poolId_ != 0U) {
136  QF_EVT_REF_CTR_INC_(e); /* increment the reference counter */
137  }
138 
139  --nFree; /* one free entry just used up */
140  me->nFree = nFree; /* update the volatile */
141  if (me->nMin > nFree) {
142  me->nMin = nFree; /* update minimum so far */
143  }
144 
145  QS_BEGIN_NOCRIT_PRE_(QS_QF_EQUEUE_POST, qs_id)
146  QS_TIME_PRE_(); /* timestamp */
147  QS_SIG_PRE_(e->sig); /* the signal of this event */
148  QS_OBJ_PRE_(me); /* this queue object */
149  QS_2U8_PRE_(e->poolId_, e->refCtr_); /* pool Id & ref Count */
150  QS_EQC_PRE_(nFree); /* number of free entries */
151  QS_EQC_PRE_(me->nMin); /* min number of free entries */
152  QS_END_NOCRIT_PRE_()
153 
154  /* was the queue empty? */
155  if (me->frontEvt == (QEvt *)0) {
156  me->frontEvt = e; /* deliver event directly */
157  }
158  /* queue was not empty, insert event into the ring-buffer */
159  else {
160  /* insert event into the ring buffer (FIFO)... */
161  QF_PTR_AT_(me->ring, me->head) = e; /* insert e into buffer */
162  /* need to wrap the head? */
163  if (me->head == 0U) {
164  me->head = me->end; /* wrap around */
165  }
166  --me->head;
167  }
168  status = true; /* event posted successfully */
169  }
170  else {
174  Q_ASSERT_CRIT_(210, margin != QF_NO_MARGIN);
175 
176  QS_BEGIN_NOCRIT_PRE_(QS_QF_EQUEUE_POST_ATTEMPT, qs_id)
177  QS_TIME_PRE_(); /* timestamp */
178  QS_SIG_PRE_(e->sig); /* the signal of this event */
179  QS_OBJ_PRE_(me); /* this queue object */
180  QS_2U8_PRE_(e->poolId_, e->refCtr_); /* pool Id & ref Count */
181  QS_EQC_PRE_(nFree); /* number of free entries */
182  QS_EQC_PRE_(margin); /* margin requested */
183  QS_END_NOCRIT_PRE_()
184 
185  status = false;
186  }
187  QF_CRIT_X_();
188 
189  return status;
190 }

284 QEvt const *QEQueue_get(QEQueue * const me, uint_fast8_t const qs_id) {
285  QEvt const *e;
286  QF_CRIT_STAT_
287 
288  (void)qs_id; /* unused parameter (outside Q_SPY build configuration) */
289 
290  QF_CRIT_E_();
291  e = me->frontEvt; /* always remove the event from the front location */
292 
293  /* was the queue not empty? */
294  if (e != (QEvt *)0) {
295  /* use a temporary variable to increment volatile me->nFree */
296  QEQueueCtr nFree = me->nFree + 1U;
297  me->nFree = nFree; /* update the number of free */
298 
299  /* any events in the ring buffer? */
300  if (nFree <= me->end) {
301  me->frontEvt = QF_PTR_AT_(me->ring, me->tail); /* get from tail */
302  if (me->tail == 0U) { /* need to wrap the tail? */
303  me->tail = me->end; /* wrap around */
304  }
305  --me->tail;
306 
307  QS_BEGIN_NOCRIT_PRE_(QS_QF_EQUEUE_GET, qs_id)
308  QS_TIME_PRE_(); /* timestamp */

310  QS_OBJ_PRE_(me); /* this queue object */

312  QS_EQC_PRE_(nFree); /* number of free entries */
313  QS_END_NOCRIT_PRE_()
314  }
315  else {
