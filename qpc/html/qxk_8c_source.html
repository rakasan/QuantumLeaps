249 QSchedStatus QXK_schedLock(uint_fast8_t ceiling) {
250  QSchedStatus stat;
251  QF_CRIT_STAT_
252  QF_CRIT_E_();
253 
257  Q_REQUIRE_ID(400, !QXK_ISR_CONTEXT_());
258 
259  /* first store the previous lock prio if below the ceiling */
260  if ((uint_fast8_t)QXK_attr_.lockPrio < ceiling) {
261  stat = ((QSchedStatus)QXK_attr_.lockPrio << 8);
262  QXK_attr_.lockPrio = (uint8_t)ceiling;
263 
264  QS_BEGIN_NOCRIT_PRE_(QS_SCHED_LOCK, 0U)
265  QS_TIME_PRE_(); /* timestamp */
266  /* the previous lock prio & new lock prio */
267  QS_2U8_PRE_(stat, QXK_attr_.lockPrio);
268  QS_END_NOCRIT_PRE_()
269 
270  /* add the previous lock holder priority */
271  stat |= (QSchedStatus)QXK_attr_.lockHolder;
272  QXK_attr_.lockHolder = (QXK_attr_.curr != (QActive *)0)
273  ? QXK_attr_.curr->prio
274  : 0U;
275  }
276  else {
277  stat = 0xFFU;
278  }
279  QF_CRIT_X_();
280 
281  return stat; /* return the status to be saved in a stack variable */

305 void QXK_schedUnlock(QSchedStatus stat) {
306  /* has the scheduler been actually locked by the last QXK_schedLock()? */
307  if (stat != 0xFFU) {
308  uint_fast8_t lockPrio = (uint_fast8_t)QXK_attr_.lockPrio;
309  uint_fast8_t prevPrio = (uint_fast8_t)(stat >> 8U);
310  QF_CRIT_STAT_
311  QF_CRIT_E_();
312 
317  Q_REQUIRE_ID(500, (!QXK_ISR_CONTEXT_())
318  && (lockPrio > prevPrio));
319 
320  QS_BEGIN_NOCRIT_PRE_(QS_SCHED_UNLOCK, 0U)
321  QS_TIME_PRE_(); /* timestamp */
322  /* prio before unlocking & prio after unlocking */
323  QS_2U8_PRE_(lockPrio, prevPrio);
324  QS_END_NOCRIT_PRE_()
325 
326  /* restore the previous lock priority and lock holder */
327  QXK_attr_.lockPrio = (uint8_t)prevPrio;
328  QXK_attr_.lockHolder = (uint8_t)(stat & 0xFFU);
329 
330  /* find the highest-prio thread ready to run */
331  if (QXK_sched_() != 0U) { /* priority found? */
332  QXK_activate_(); /* activate any unlocked basic threads */
333  }
334 
335  QF_CRIT_X_();
336  }

353 uint_fast8_t QXK_sched_(void) {
354  uint_fast8_t p; /* next thread priority */
355  QActive *next;
356 
357  /* find the highest-prio thread ready to run */
358  QPSet_findMax(&QXK_attr_.readySet, p);
359 
360  if (p <= (uint_fast8_t)QXK_attr_.lockPrio) { /* below the lock prio? */
361  /* dynamic priority of the thread holding the lock */
362  p = (uint_fast8_t)QF_active_[QXK_attr_.lockHolder]->dynPrio;
363  if (p != 0U) {
364  Q_ASSERT_ID(610, QPSet_hasElement(&QXK_attr_.readySet, p));
365  }
366  }
367  next = QF_active_[p];
368 
369  /* the thread found must be registered in QF */
370  Q_ASSERT_ID(620, next != (QActive *)0);
371 
372  /* is the current thread a basic-thread? */
373  if (QXK_attr_.curr == (QActive *)0) {
374 
375  /* is next a basic-thread? */
376  if (next->osObject == (void *)0) {
377  if (p > (uint_fast8_t)QXK_attr_.actPrio) {
378  QXK_attr_.next = next; /* set the next AO to activate */
379  }
380  else {
381  QXK_attr_.next = (QActive *)0;
382  p = 0U; /* no activation needed */
383  }
384  }
385  else { /* this is an extened-thread */
386 
387  QS_BEGIN_NOCRIT_PRE_(QS_SCHED_NEXT, next->prio)
388  QS_TIME_PRE_(); /* timestamp */
389  /* prio of the next AO & prio of the curr AO */
390  QS_2U8_PRE_(p, QXK_attr_.actPrio);
391  QS_END_NOCRIT_PRE_()
392 
393  QXK_attr_.next = next;
394  p = 0U; /* no activation needed */
395  QXK_CONTEXT_SWITCH_();
396  }
397  }
398  else { /* currently executing an extended-thread */
399 
400  /* is the next thread different from the current? */
401  if (next != QXK_attr_.curr) {
402 
403  QS_BEGIN_NOCRIT_PRE_(QS_SCHED_NEXT, next->prio)
404  QS_TIME_PRE_(); /* timestamp */
405  /* next prio & current prio */
406  QS_2U8_PRE_(p, QXK_attr_.curr->prio);
407  QS_END_NOCRIT_PRE_()
408 
409  QXK_attr_.next = next;
410  p = 0U; /* no activation needed */
411  QXK_CONTEXT_SWITCH_();
412  }
413  else { /* next is the same as the current */
414  QXK_attr_.next = (QActive *)0; /* no need to context-switch */
415  p = 0U; /* no activation needed */
416  }
417  }
418  return p;
419 }

431 void QXK_activate_(void) {
432  uint_fast8_t const pin = (uint_fast8_t)QXK_attr_.actPrio;
433  QActive *a = QXK_attr_.next; /* the next AO (basic-thread) to run */
434  uint_fast8_t p;
435 
436  /* QXK Context switch callback defined or QS tracing enabled? */
437 #if (defined QXK_ON_CONTEXT_SW) || (defined Q_SPY)
438  uint_fast8_t pprev = pin;
439 #endif /* QXK_ON_CONTEXT_SW || Q_SPY */
440 
442  Q_REQUIRE_ID(700, (a != (QActive *)0) && (pin < QF_MAX_ACTIVE));
443 
444  /* dynamic priority of the next AO */
445  p = (uint_fast8_t)a->dynPrio; /* dynamic priority of the next AO */
446 
447  /* loop until no more ready-to-run AOs of higher prio than the initial */
448  do {
449  QEvt const *e;
450 
451  a = QF_active_[p]; /* obtain the pointer to the AO */
452  QXK_attr_.actPrio = (uint8_t)p; /* this becomes the base prio */
453  QXK_attr_.next = (QActive *)0; /* clear the next AO */
454 
455  QS_BEGIN_NOCRIT_PRE_(QS_SCHED_NEXT, a->prio)
456  QS_TIME_PRE_(); /* timestamp */
457  /* next prio & prev prio */
458 
459  QS_2U8_PRE_(p, pprev);
460  QS_END_NOCRIT_PRE_()
461 
462 #if (defined QXK_ON_CONTEXT_SW) || (defined Q_SPY)
463  if (p != pprev) { /* changing threads? */
464 
465 #ifdef QXK_ON_CONTEXT_SW
466  Q_ASSERT_ID(710, pprev < QF_MAX_ACTIVE);
467 
468  /* context-switch callback */
469  QXK_onContextSw(((pprev != 0U) ? QF_active_[pprev] : (QActive*)0),
470  a);
471 #endif /* QXK_ON_CONTEXT_SW */
472 
473  pprev = p; /* update previous priority */
474  }
475 #endif /* QXK_ON_CONTEXT_SW || Q_SPY */
476 
477  QF_INT_ENABLE(); /* unconditionally enable interrupts */
478 
479  /* perform the run-to-completion (RTC) step...
480  * 1. retrieve the event from the AO's event queue, which by this
481  * time must be non-empty and QActive_get_() asserts it.
482  * 2. dispatch the event to the AO's state machine.
483  * 3. determine if event is garbage and collect it if so
484  */
485  e = QActive_get_(a);
486  QHSM_DISPATCH(&a->super, e, a->prio);
487  QF_gc(e);
488 
489  QF_INT_DISABLE(); /* unconditionally disable interrupts */
490 
491  if (a->eQueue.frontEvt == (QEvt *)0) { /* empty queue? */
492  QPSet_remove(&QXK_attr_.readySet, p);
493  }
494 
495  /* find new highest-prio AO ready to run...
496  * NOTE: this part must match the QXK_sched_(),
497  * current is a basic-thread path.
498  */
499  QPSet_findMax(&QXK_attr_.readySet, p);
500 
501  if (p <= (uint_fast8_t)QXK_attr_.lockPrio) { /* below lock prio? */
502  p = (uint_fast8_t)QXK_attr_.lockHolder; /* thread holding lock */
503  if (p != 0U) {
504  Q_ASSERT_ID(710, QPSet_hasElement(&QXK_attr_.readySet, p));
505  }
506  }
507  a = QF_active_[p];
508 
509  /* the AO must be registered in QF */
510  Q_ASSERT_ID(720, a != (QActive *)0);
511 
512  /* is the next a basic thread? */
513  if (a->osObject == (void *)0) {
514  if (p > pin) {
515  QXK_attr_.next = a;
516  }
517  else {
518  QXK_attr_.next = (QActive *)0;
519  p = 0U; /* no activation needed */
520  }
521  }
522  else { /* next is the extened thread */
523 
524  QS_BEGIN_NOCRIT_PRE_(QS_SCHED_NEXT, a->prio)
525  QS_TIME_PRE_(); /* timestamp */
526  /* next prio & curr prio */
527  QS_2U8_PRE_(p, QXK_attr_.actPrio);
528  QS_END_NOCRIT_PRE_()
529 
530  QXK_attr_.next = a;
531  p = 0U; /* no activation needed */
532  QXK_CONTEXT_SWITCH_();
533  }
534  } while (p != 0U); /* while activation needed */
535 
536  QXK_attr_.actPrio = (uint8_t)pin; /* restore the base prio */
537 
538 #if (defined QXK_ON_CONTEXT_SW) || (defined Q_SPY)
539  if (pin != 0U) { /* resuming an active object? */
540  a = QF_active_[pin]; /* the pointer to the preempted AO */
541 
542  QS_BEGIN_NOCRIT_PRE_(QS_SCHED_RESUME, a->prio)
543  QS_TIME_PRE_(); /* timestamp */
544  QS_2U8_PRE_(pin, pprev); /* resumed prio & previous prio */
545  QS_END_NOCRIT_PRE_()
546  }
547  else { /* resuming priority==0 --> idle */
548  a = (QActive *)0;
549 
550  QS_BEGIN_NOCRIT_PRE_(QS_SCHED_IDLE, 0U)
551  QS_TIME_PRE_(); /* timestamp */
552  QS_U8_PRE_(pprev); /* previous priority */
553  QS_END_NOCRIT_PRE_()
554  }
555 
556 #ifdef QXK_ON_CONTEXT_SW
557  /* context-switch callback */
558  QXK_onContextSw(QF_active_[pprev], a);
559 #endif /* QXK_ON_CONTEXT_SW */
560 
561 #endif /* QXK_ON_CONTEXT_SW || Q_SPY */
562 }
