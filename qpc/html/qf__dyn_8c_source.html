164 QEvt *QF_newX_(uint_fast16_t const evtSize,
165  uint_fast16_t const margin, enum_t const sig)
166 {
167  QEvt *e;
168  uint_fast8_t idx;
169  QS_CRIT_STAT_
170 
171  /* find the pool index that fits the requested event size ... */
172  for (idx = 0U; idx < QF_maxPool_; ++idx) {
173  if (evtSize <= QF_EPOOL_EVENT_SIZE_(QF_pool_[idx])) {
174  break;
175  }
176  }
177  /* cannot run out of registered pools */
178  Q_ASSERT_ID(310, idx < QF_maxPool_);
179 
180  /* get e -- platform-dependent */
181 #ifdef Q_SPY
182  QF_EPOOL_GET_(QF_pool_[idx], e,
183  ((margin != QF_NO_MARGIN) ? margin : 0U),
184  (uint_fast8_t)QS_EP_ID + idx + 1U);
185 #else
186  QF_EPOOL_GET_(QF_pool_[idx], e,
187  ((margin != QF_NO_MARGIN) ? margin : 0U), 0U);
188 #endif
189 
190  /* was e allocated correctly? */
191  if (e != (QEvt *)0) {
192  e->sig = (QSignal)sig; /* set signal for this event */
193  e->poolId_ = (uint8_t)(idx + 1U); /* store the pool ID */
194  e->refCtr_ = 0U; /* set the reference counter to 0 */
195 
196  QS_BEGIN_PRE_(QS_QF_NEW, (uint_fast8_t)QS_EP_ID + e->poolId_)
197  QS_TIME_PRE_(); /* timestamp */
198  QS_EVS_PRE_(evtSize); /* the size of the event */
199  QS_SIG_PRE_(sig); /* the signal of the event */
200  QS_END_PRE_()
201  }
202  /* event cannot be allocated */
203  else {
204  /* must tolerate failed allocation */
205  Q_ASSERT_ID(320, margin != QF_NO_MARGIN);
206 
207  QS_BEGIN_PRE_(QS_QF_NEW_ATTEMPT, (uint_fast8_t)QS_EP_ID + idx + 1U)
208  QS_TIME_PRE_(); /* timestamp */
209  QS_EVS_PRE_(evtSize); /* the size of the event */
210  QS_SIG_PRE_(sig); /* the signal of the event */
211  QS_END_PRE_()
212  }
213  return e; /* can't be NULL if we can't tolerate failed allocation */
214 }

240 void QF_gc(QEvt const * const e) {
241 
242  /* is it a dynamic event? */
243  if (e->poolId_ != 0U) {
244  QF_CRIT_STAT_
245  QF_CRIT_E_();
246 
247  /* isn't this the last reference? */
248  if (e->refCtr_ > 1U) {
249 
250  QS_BEGIN_NOCRIT_PRE_(QS_QF_GC_ATTEMPT,
251  (uint_fast8_t)QS_EP_ID + e->poolId_)
252  QS_TIME_PRE_(); /* timestamp */
253  QS_SIG_PRE_(e->sig); /* the signal of the event */
254  QS_2U8_PRE_(e->poolId_, e->refCtr_); /* pool Id & ref Count */
255  QS_END_NOCRIT_PRE_()
256 
257  QF_EVT_REF_CTR_DEC_(e); /* decrement the ref counter */
258 
259  QF_CRIT_X_();
260  }
261  /* this is the last reference to this event, recycle it */
262  else {
263  uint_fast8_t idx = (uint_fast8_t)e->poolId_ - 1U;
264 
265  QS_BEGIN_NOCRIT_PRE_(QS_QF_GC,
266  (uint_fast8_t)QS_EP_ID + e->poolId_)
267  QS_TIME_PRE_(); /* timestamp */
268  QS_SIG_PRE_(e->sig); /* the signal of the event */
269  QS_2U8_PRE_(e->poolId_, e->refCtr_); /* pool Id & ref Count */
270  QS_END_NOCRIT_PRE_()
271 
272  QF_CRIT_X_();
273 
274  /* pool ID must be in range */
275  Q_ASSERT_ID(410, idx < QF_maxPool_);
276 
277  /* cast 'const' away, which is OK, because it's a pool event */
278 #ifdef Q_SPY
279  QF_EPOOL_PUT_(QF_pool_[idx], QF_EVT_CONST_CAST_(e),
280  (uint_fast8_t)QS_EP_ID + e->poolId_);
281 #else
282  QF_EPOOL_PUT_(QF_pool_[idx], QF_EVT_CONST_CAST_(e), 0U);
283 #endif
284  }
285  }
