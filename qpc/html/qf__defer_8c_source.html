119 bool QActive_recall(QActive * const me, QEQueue * const eq) {
120  QEvt const *e = QEQueue_get(eq, me->prio);
121  bool recalled;
122 
123  /* event available? */
124  if (e != (QEvt *)0) {
125  QF_CRIT_STAT_
126 
127  QACTIVE_POST_LIFO(me, e); /* post it to the front of the AO's queue */
128 
129  QF_CRIT_E_();
130 
131  /* is it a dynamic event? */
132  if (e->poolId_ != 0U) {
133 
134  /* after posting to the AO's queue the event must be referenced
135  * at least twice: once in the deferred event queue (eq->get()
136  * did NOT decrement the reference counter) and once in the
137  * AO's event queue.
138  */
139  Q_ASSERT_CRIT_(210, e->refCtr_ >= 2U);
140 
141  /* we need to decrement the reference counter once, to account
142  * for removing the event from the deferred event queue.
143  */
144  QF_EVT_REF_CTR_DEC_(e); /* decrement the reference counter */
145  }
146 
147  QS_BEGIN_NOCRIT_PRE_(QS_QF_ACTIVE_RECALL, me->prio)
148  QS_TIME_PRE_(); /* time stamp */
149  QS_OBJ_PRE_(me); /* this active object */
150  QS_OBJ_PRE_(eq); /* the deferred queue */
151  QS_SIG_PRE_(e->sig); /* the signal of the event */
152  QS_2U8_PRE_(e->poolId_, e->refCtr_); /* pool Id & ref Count */
153  QS_END_NOCRIT_PRE_()
154 
155  QF_CRIT_X_();
156  recalled = true;
157  }
158  else {
159  QS_CRIT_STAT_
160 
161  QS_BEGIN_PRE_(QS_QF_ACTIVE_RECALL_ATTEMPT, me->prio)
162  QS_TIME_PRE_(); /* time stamp */
