119 void QS_glbFilter_(int_fast16_t const filter) {
120  bool isRemove = (filter < 0);
121  uint8_t rec = isRemove ? (uint8_t)(-filter) : (uint8_t)filter;
122  switch (rec) {
123  case QS_ALL_RECORDS: {
124  uint8_t tmp = (isRemove ? 0x00U : 0xFFU);
125  uint_fast8_t i;
126  /* set all global filters (partially unrolled loop) */
127  for (i = 0U; i < Q_DIM(QS_priv_.glbFilter); i += 4U) {
128  QS_priv_.glbFilter[i ] = tmp;
129  QS_priv_.glbFilter[i + 1U] = tmp;
130  QS_priv_.glbFilter[i + 2U] = tmp;
131  QS_priv_.glbFilter[i + 3U] = tmp;
132  }
133  if (isRemove) {
134  /* leave the "not maskable" filters enabled,
135  * see qs.h, Miscellaneous QS records (not maskable)
136  */
137  QS_priv_.glbFilter[0] = 0x01U;
138  QS_priv_.glbFilter[7] = 0xFCU;
139  QS_priv_.glbFilter[8] = 0x7FU;
140  }
141  else {
142  /* never turn the last 3 records on (0x7D, 0x7E, 0x7F) */
143  QS_priv_.glbFilter[15] = 0x1FU;
144  }
145  break;
146  }
147  case QS_SM_RECORDS:
148  if (isRemove) {
149  QS_priv_.glbFilter[0] &= (uint8_t)(~0xFEU & 0xFFU);
150  QS_priv_.glbFilter[1] &= (uint8_t)(~0x03U & 0xFFU);
151  QS_priv_.glbFilter[6] &= (uint8_t)(~0x80U & 0xFFU);
152  QS_priv_.glbFilter[7] &= (uint8_t)(~0x03U & 0xFFU);
153  }
154  else {
155  QS_priv_.glbFilter[0] |= 0xFEU;
156  QS_priv_.glbFilter[1] |= 0x03U;
157  QS_priv_.glbFilter[6] |= 0x80U;
158  QS_priv_.glbFilter[7] |= 0x03U;
159  }
160  break;
161  case QS_AO_RECORDS:
162  if (isRemove) {
163  QS_priv_.glbFilter[1] &= (uint8_t)(~0xFCU & 0xFFU);
164  QS_priv_.glbFilter[2] &= (uint8_t)(~0x07U & 0xFFU);
165  QS_priv_.glbFilter[5] &= (uint8_t)(~0x20U & 0xFFU);
166  }
167  else {
168  QS_priv_.glbFilter[1] |= 0xFCU;
169  QS_priv_.glbFilter[2] |= 0x07U;
170  QS_priv_.glbFilter[5] |= 0x20U;
171  }
172  break;
173  case QS_EQ_RECORDS:
174  if (isRemove) {
175  QS_priv_.glbFilter[2] &= (uint8_t)(~0x78U & 0xFFU);
176  QS_priv_.glbFilter[5] &= (uint8_t)(~0x40U & 0xFFU);
177  }
178  else {

180  QS_priv_.glbFilter[5] |= 0x40U;
181  }
182  break;
183  case QS_MP_RECORDS:

186  QS_priv_.glbFilter[5] &= (uint8_t)(~0x80U & 0xFFU);

263  else {
264  QS_priv_.glbFilter[15] |= 0x1FU;
265  }
266  break;
267  case QS_UA_RECORDS:
268  if (isRemove) {
269  QS_priv_.glbFilter[12] &= (uint8_t)(~0xF0U & 0xFFU);
270  QS_priv_.glbFilter[13] = 0U;
271  QS_priv_.glbFilter[14] = 0U;
272  QS_priv_.glbFilter[15] &= (uint8_t)(~0x1FU & 0xFFU);
273  }
274  else {
275  QS_priv_.glbFilter[12] |= 0xF0U;
276  QS_priv_.glbFilter[13] |= 0xFFU;
277  QS_priv_.glbFilter[14] |= 0xFFU;
278  QS_priv_.glbFilter[15] |= 0x1FU;
279  }
280  break;
281  default:
282  /* QS rec number can't exceed 0x7D, so no need for escaping */
283  Q_ASSERT_ID(210, rec < 0x7DU);
284 
285  if (isRemove) {
286  QS_priv_.glbFilter[rec >> 3U]
287  &= (uint8_t)(~(1U << (rec & 7U)) & 0xFFU);
288  }
289  else {
290  QS_priv_.glbFilter[rec >> 3U]
291  |= (1U << (rec & 7U));
292  /* never turn the last 3 records on (0x7D, 0x7E, 0x7F) */
293  QS_priv_.glbFilter[15] &= 0x1FU;
294  }
295  break;
296  }

314 void QS_locFilter_(int_fast16_t const filter) {
315  bool isRemove = (filter < 0);
316  uint8_t qs_id = isRemove ? (uint8_t)(-filter) : (uint8_t)filter;
317  uint8_t tmp = (isRemove ? 0x00U : 0xFFU);
318  uint_fast8_t i;
319  switch (qs_id) {
320  case QS_ALL_IDS:
321  /* set all global filters (partially unrolled loop) */
322  for (i = 0U; i < Q_DIM(QS_priv_.locFilter); i += 4U) {
323  QS_priv_.locFilter[i ] = tmp;
324  QS_priv_.locFilter[i + 1U] = tmp;
325  QS_priv_.locFilter[i + 2U] = tmp;
326  QS_priv_.locFilter[i + 3U] = tmp;
327  }
328  break;
329  case QS_AO_IDS:
330  for (i = 0U; i < 8U; i += 4U) {
331  QS_priv_.locFilter[i ] = tmp;
332  QS_priv_.locFilter[i + 1U] = tmp;
333  QS_priv_.locFilter[i + 2U] = tmp;
334  QS_priv_.locFilter[i + 3U] = tmp;
335  }
336  break;
337  case QS_EP_IDS:
338  i = 8U;
339  QS_priv_.locFilter[i ] = tmp;
340  QS_priv_.locFilter[i + 1U] = tmp;
341  break;
342  case QS_AP_IDS:
343  i = 12U;
344  QS_priv_.locFilter[i ] = tmp;
345  QS_priv_.locFilter[i + 1U] = tmp;
346  QS_priv_.locFilter[i + 2U] = tmp;
347  QS_priv_.locFilter[i + 3U] = tmp;
348  break;
349  default:
350  if (qs_id < 0x7FU) {
351  if (isRemove) {
352  QS_priv_.locFilter[qs_id >> 3U]
353  &= (uint8_t)(~(1U << (qs_id & 7U)) & 0xFFU);
354  }
355  else {
356  QS_priv_.locFilter[qs_id >> 3U]
357  |= (1U << (qs_id & 7U));
358  }
359  }
360  else {
361  Q_ERROR_ID(310); /* incorrect qs_id */
362  }
363  break;
364  }
365  QS_priv_.locFilter[0] |= 0x01U; /* leave QS_ID == 0 always on */
366 }

376 void QS_beginRec_(uint_fast8_t rec) {
377  uint8_t b = (uint8_t)(QS_priv_.seq + 1U);
378  uint8_t chksum = 0U; /* reset the checksum */
379  uint8_t *buf = QS_priv_.buf; /* put in a temporary (register) */
380  QSCtr head = QS_priv_.head; /* put in a temporary (register) */
381  QSCtr end = QS_priv_.end; /* put in a temporary (register) */
382 
383  QS_priv_.seq = b; /* store the incremented sequence num */
384  QS_priv_.used += 2U; /* 2 bytes about to be added */
385 
386  QS_INSERT_ESC_BYTE_(b)
387 
388  chksum = (uint8_t)(chksum + rec); /* update checksum */
389  QS_INSERT_BYTE_((uint8_t)rec) /* rec byte does not need escaping */
390 
391  QS_priv_.head = head; /* save the head */
392  QS_priv_.chksum = chksum; /* save the checksum */
393 }

403 void QS_endRec_(void) {
404  uint8_t *buf = QS_priv_.buf; /* put in a temporary (register) */
405  QSCtr head = QS_priv_.head;
406  QSCtr end = QS_priv_.end;
407  uint8_t b = QS_priv_.chksum;
408  b ^= 0xFFU; /* invert the bits in the checksum */
409 
410  QS_priv_.used += 2U; /* 2 bytes about to be added */
411 
412  if ((b != QS_FRAME) && (b != QS_ESC)) {
413  QS_INSERT_BYTE_(b)
414  }
415  else {
416  QS_INSERT_BYTE_(QS_ESC)
417  QS_INSERT_BYTE_(b ^ QS_ESC_XOR)
418  ++QS_priv_.used; /* account for the ESC byte */
419  }
420 
421  QS_INSERT_BYTE_(QS_FRAME) /* do not escape this QS_FRAME */
422 
423  QS_priv_.head = head; /* save the head */
424 
425  /* overrun over the old data? */
426  if (QS_priv_.used > end) {
427  QS_priv_.used = end; /* the whole buffer is used */
428  QS_priv_.tail = head; /* shift the tail to the old data */
429  }

747 uint16_t QS_getByte(void) {
748  uint16_t ret;
749  if (QS_priv_.used == 0U) {
750  ret = QS_EOD; /* set End-Of-Data */
751  }
752  else {
753  uint8_t const *buf = QS_priv_.buf; /* put in a temporary */
754  QSCtr tail = QS_priv_.tail; /* put in a temporary (register) */
755  ret = (uint16_t)buf[tail]; /* set the byte to return */
756  ++tail; /* advance the tail */
757  if (tail == QS_priv_.end) { /* tail wrap around? */
758  tail = 0U;
759  }
760  QS_priv_.tail = tail; /* update the tail */
761  --QS_priv_.used; /* one less byte used */
762  }
763  return ret; /* return the byte or EOD */
764 }

766 /****************************************************************************/
789 uint8_t const *QS_getBlock(uint16_t *pNbytes) {
790  QSCtr used = QS_priv_.used; /* put in a temporary (register) */
791  uint8_t const *buf;
792 
793  /* any bytes used in the ring buffer? */
794  if (used != 0U) {
795  QSCtr tail = QS_priv_.tail; /* put in a temporary (register) */
796  QSCtr end = QS_priv_.end; /* put in a temporary (register) */
797  QSCtr n = (QSCtr)(end - tail);
798  if (n > used) {
799  n = used;
800  }
801  if (n > (QSCtr)(*pNbytes)) {
802  n = (QSCtr)(*pNbytes);
803  }
804  *pNbytes = (uint16_t)n; /* n-bytes available */
805  buf = &QS_priv_.buf[tail]; /* the bytes are at the tail */
806 
807  QS_priv_.used = (QSCtr)(used - n);
808  tail += n;
809  if (tail == end) {
810  tail = 0U;
811  }
812  QS_priv_.tail = tail;
813  }
814 
815  else { /* no bytes available */
816  *pNbytes = 0U; /* no bytes available right now */
817  buf = (uint8_t *)0; /* no bytes available right now */
818  }
819  return buf;
820 }
