134 bool QXMutex_lock(QXMutex * const me,
135  uint_fast16_t const nTicks)
136 {
137  bool locked = true; /* assume that the mutex will be locked */
138  QXThread *curr;
139  QF_CRIT_STAT_
140 
141  QF_CRIT_E_();
142  curr = QXK_PTR_CAST_(QXThread*, QXK_attr_.curr);
143 
152  Q_REQUIRE_ID(200, (!QXK_ISR_CONTEXT_()) /* don't call from an ISR! */
153  && (curr != (QXThread *)0) /* current thread must be extended */
154  && ((me->ceiling == 0U)
155  || (curr->super.prio < me->ceiling)) /* below ceiling */
156  && (me->ceiling <= QF_MAX_ACTIVE) /* in range */
157  && (curr->super.super.temp.obj == (QMState *)0)); /* not blocked */
159  Q_REQUIRE_ID(201, QXK_attr_.lockHolder != curr->super.prio);
160 
161  /* is the mutex available? */
162  if (me->lockNest == 0U) {
163  me->lockNest = 1U;
164 
165  if (me->ceiling != 0U) {
166  /* the priority slot must be occupied by this mutex */
167  Q_ASSERT_ID(210, QF_active_[me->ceiling]
168  == QXK_PTR_CAST_(QActive*, me));
169 
170  /* boost the dynamic priority of this thread to the ceiling */
171  QPSet_remove(&QXK_attr_.readySet,
172  (uint_fast8_t)curr->super.dynPrio);
173  curr->super.dynPrio = me->ceiling;
174  QPSet_insert(&QXK_attr_.readySet,
175  (uint_fast8_t)curr->super.dynPrio);
176  QF_active_[me->ceiling] = &curr->super;
177  }
178 
179  /* make the curr thread the new mutex holder */
180  me->holderPrio = (uint8_t)curr->super.prio;
181 
182  QS_BEGIN_NOCRIT_PRE_(QS_MUTEX_LOCK, curr->super.prio)
183  QS_TIME_PRE_(); /* timestamp */
184  QS_2U8_PRE_(curr->super.prio, /* the start prio */
185  me->ceiling); /* current ceiling */
186  QS_END_NOCRIT_PRE_()
187  }
188  /* is the mutex locked by this thread already (nested locking)? */
189  else if (me->holderPrio == curr->super.prio) {
190 
191  /* the nesting level must not exceed the dynamic range of uint8_t */
192  Q_ASSERT_ID(220, me->lockNest < 0xFFU);
193 
194  ++me->lockNest;
195  }
196  else { /* the mutex is alredy locked by a different thread */
197  uint_fast8_t p;
198 
199  /* the ceiling holder priority must be valid */
200  Q_ASSERT_ID(230, 0U < me->holderPrio);
201  Q_ASSERT_ID(231, me->holderPrio <= QF_MAX_ACTIVE);
202 
203  if (me->ceiling != 0U) {
204  /* the prio slot must be occupied by the thr. holding the mutex */
205  Q_ASSERT_ID(240, QF_active_[me->ceiling]
206  == QF_active_[me->holderPrio]);
207  }
208 
209  /* remove the curr dynamic prio from the ready set (block)
210  * and insert it to the waiting set on this mutex
211  */
212  p = (uint_fast8_t)curr->super.dynPrio;
213  QPSet_remove(&QXK_attr_.readySet, p);
214  QPSet_insert(&me->waitSet, p);
215 
216  /* store the blocking object (this mutex) */
217  curr->super.super.temp.obj = QXK_PTR_CAST_(QMState*, me);
218  QXThread_teArm_(curr, (QSignal)QXK_SEMA_SIG, nTicks);
219 
220  /* schedule the next thread if multitasking started */
221  (void)QXK_sched_();
222  QF_CRIT_X_();
223  QF_CRIT_EXIT_NOP(); /* BLOCK here !!! */
224 
225  QF_CRIT_E_(); /* AFTER unblocking... */
226  /* the blocking object must be this mutex */
227  Q_ASSERT_ID(240, curr->super.super.temp.obj
228  == QXK_PTR_CAST_(QMState*, me));
229 
230  /* did the blocking time-out? (signal of zero means that it did) */
231  if (curr->timeEvt.super.sig == 0U) {
232  if (QPSet_hasElement(&me->waitSet, p)) { /* still waiting? */
233  QPSet_remove(&me->waitSet, p); /* remove unblocked thread */
234  locked = false; /* the mutex was NOT locked */
235  }
236  }
237  else { /* blocking did NOT time out */
238  /* the thread must NOT be waiting on this mutex */
239  Q_ASSERT_ID(250, !QPSet_hasElement(&me->waitSet, p));
240  }
241  curr->super.super.temp.obj = (QMState *)0; /* clear blocking obj. */
242  }
243  QF_CRIT_X_();
244 
245  return locked;
246 }

269 bool QXMutex_tryLock(QXMutex * const me) {
270  QActive *curr;
271  QF_CRIT_STAT_
272 
273  QF_CRIT_E_();
274  curr = QXK_attr_.curr;
275  if (curr == (QActive *)0) { /* called from a basic thread? */
276  curr = QF_active_[QXK_attr_.actPrio];
277  }
278 
286  Q_REQUIRE_ID(300, (!QXK_ISR_CONTEXT_()) /* don't call from an ISR! */
287  && (curr != (QActive *)0) /* current thread must be valid */
288  && ((me->ceiling == 0U) /* ceiling not used */
289  || (curr->prio < me->ceiling)) /* below ceiling */
290  && (me->ceiling <= QF_MAX_ACTIVE));
292  Q_REQUIRE_ID(301, QXK_attr_.lockHolder != curr->prio);
293 
294  /* is the mutex available? */
295  if (me->lockNest == 0U) {
296  me->lockNest = 1U;
297 
298  if (me->ceiling != 0U) {
299  /* the priority slot must be set to this mutex */
300  Q_ASSERT_ID(310, QF_active_[me->ceiling]
301  == QXK_PTR_CAST_(QActive*, me));
302 
303  /* boost the dynamic priority of this thread to the ceiling */
304  QPSet_remove(&QXK_attr_.readySet, (uint_fast8_t)curr->dynPrio);
305  curr->dynPrio = me->ceiling;
306  QPSet_insert(&QXK_attr_.readySet, (uint_fast8_t)curr->dynPrio);
307  QF_active_[me->ceiling] = curr;
308  }
309 
310  /* make curr thread the new mutex holder */
311  me->holderPrio = curr->prio;
312 
313  QS_BEGIN_NOCRIT_PRE_(QS_MUTEX_LOCK, curr->prio)
314  QS_TIME_PRE_(); /* timestamp */
315  QS_2U8_PRE_(curr->prio, me->ceiling);
316  QS_END_NOCRIT_PRE_()
317  }
318  /* is the mutex held by this thread already (nested locking)? */
319  else if (me->holderPrio == curr->prio) {
320  /* the nesting level must not exceed the dynamic range of uint8_t */
321  Q_ASSERT_ID(320, me->lockNest < 0xFFU);
322 
323  ++me->lockNest;
324  }
325  else { /* the mutex is alredy locked by a different thread */
326  if (me->ceiling != 0U) {
327  /* the prio slot must be claimed by the mutex holder */
328  Q_ASSERT_ID(330, QF_active_[me->ceiling]
329  == QF_active_[me->holderPrio]);
330  }
331  curr = (QActive *)0; /* means that mutex is NOT available */
332  }
333  QF_CRIT_X_();
334 
335  return (curr != (QActive *)0);
336 }

358 void QXMutex_unlock(QXMutex * const me) {
359  QActive *curr;
360  QF_CRIT_STAT_
361 
362  QF_CRIT_E_();
363  curr = QXK_attr_.curr;
364  if (curr == (QActive *)0) { /* called from a basic thread? */
365  curr = QF_active_[QXK_attr_.actPrio];
366  }
367 
375  Q_REQUIRE_ID(400, (!QXK_ISR_CONTEXT_()) /* don't call from an ISR! */
376  && (curr != (QActive *)0) /* current thread must be valid */
377  && ((me->ceiling == 0U) /* ceiling not used OR... */
378  || (curr->dynPrio == me->ceiling)) /* curr at ceiling prio */
379  && (me->ceiling <= QF_MAX_ACTIVE));
381  Q_REQUIRE_ID(401, me->lockNest > 0U);
383  Q_REQUIRE_ID(402, me->holderPrio == curr->prio);
384 
385  /* is this the last nesting level? */
386  if (me->lockNest == 1U) {
387 
388  if (me->ceiling != 0U) {
389  /* restore the holding thread's dynamic priority to the original */
390  QPSet_remove(&QXK_attr_.readySet, (uint_fast8_t)curr->dynPrio);
391  curr->dynPrio = curr->prio;
392  QPSet_insert(&QXK_attr_.readySet, (uint_fast8_t)curr->dynPrio);
393 
394  /* put the mutex at the priority ceiling slot */
395  QF_active_[me->ceiling] = QXK_PTR_CAST_(QActive*, me);
396  }
397 
398  /* the mutex no longer held by a thread */
399  me->holderPrio = 0U;
400 
401  QS_BEGIN_NOCRIT_PRE_(QS_MUTEX_UNLOCK, curr->prio)
402  QS_TIME_PRE_(); /* timestamp */
403  /* start prio & the mutex ceiling */
404  QS_2U8_PRE_(curr->prio, me->ceiling);
405  QS_END_NOCRIT_PRE_()
406 
407  /* are any other threads waiting for this mutex? */
408  if (QPSet_notEmpty(&me->waitSet)) {
409  uint_fast8_t p;
410  QXThread *thr;
411 
412  /* find the highest-priority thread waiting on this mutex */
413  QPSet_findMax(&me->waitSet, p);
414  thr = QXK_PTR_CAST_(QXThread*, QF_active_[p]);
415 
416  /* the waiting thread must:
417  * - the ceiling must not be used; or if used
418  * - the thread must have priority below the ceiling
419  * - be registered in QF
420  * - have the dynamic priority the same as initial priority
421  * - be blocked on this mutex
422  */
423  Q_ASSERT_ID(410,
424  ((me->ceiling == 0U) || (p < (uint_fast8_t)me->ceiling))
425  && (thr != (QXThread *)0)
426  && (thr->super.dynPrio == thr->super.prio)
427  && (thr->super.super.temp.obj
428  == QXK_PTR_CAST_(QMState*, me)));
429 
430  /* disarm the internal time event */
431  (void)QXThread_teDisarm_(thr);
432 
433  if (me->ceiling != 0U) {
434  /* boost the dynamic priority of this thread to the ceiling */
435  thr->super.dynPrio = me->ceiling;
436  QF_active_[me->ceiling] = &thr->super;
437  }
438 
439  /* make the thread the new mutex holder */
440  me->holderPrio = (uint8_t)thr->super.prio;
441 
442  /* make the thread ready to run (at the ceiling prio)
443  * and remove from the waiting list */
444  QPSet_insert(&QXK_attr_.readySet,
445  (uint_fast8_t)thr->super.dynPrio);
446  QPSet_remove(&me->waitSet, p);
447 
448  QS_BEGIN_NOCRIT_PRE_(QS_MUTEX_LOCK, thr->super.prio)
449  QS_TIME_PRE_(); /* timestamp */
450  /* start priority & ceiling priority */
451  QS_2U8_PRE_(thr->super.prio, me->ceiling);
452  QS_END_NOCRIT_PRE_()
453  }
454  else { /* no threads are waiting for this mutex */
455  me->lockNest = 0U;
456 
457  if (me->ceiling != 0U) {
458  /* put the mutex at the priority ceiling slot */
459  QF_active_[me->ceiling] = QXK_PTR_CAST_(QActive*, me);
460  }
461  }
462 
463  /* schedule the next thread if multitasking started */
464  if (QXK_sched_() != 0U) {
465  QXK_activate_(); /* activate a basic thread */
466  }
467  }
468  else { /* releasing the mutex */
469  --me->lockNest; /* release one level */
470  }
471  QF_CRIT_X_();
472 }
