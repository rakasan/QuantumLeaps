118 #ifndef Q_SPY
119 void QF_publish_(QEvt const * const e)
120 #else
121 void QF_publish_(QEvt const * const e, void const * const sender)
122 #endif
123 {
124  QPSet subscrList; /* local, modifiable copy of the subscriber list */
125  QF_CRIT_STAT_
126 
128  Q_REQUIRE_ID(200, e->sig < (QSignal)QF_maxPubSignal_);
129 
130  QF_CRIT_E_();
131 
132  QS_BEGIN_NOCRIT_PRE_(QS_QF_PUBLISH, 0U)
133  QS_TIME_PRE_(); /* the timestamp */
134  QS_OBJ_PRE_(sender); /* the sender object */
135  QS_SIG_PRE_(e->sig); /* the signal of the event */
136  QS_2U8_PRE_(e->poolId_, e->refCtr_);/* pool Id & ref Count */
137  QS_END_NOCRIT_PRE_()
138 
139  /* is it a dynamic event? */
140  if (e->poolId_ != 0U) {
141  /* NOTE: The reference counter of a dynamic event is incremented to
142  * prevent premature recycling of the event while the multicasting
143  * is still in progress. At the end of the function, the garbage
144  * collector step (QF_gc()) decrements the reference counter and
145  * recycles the event if the counter drops to zero. This covers the
146  * case when the event was published without any subscribers.
147  */
148  QF_EVT_REF_CTR_INC_(e);
149  }
150 
151  /* make a local, modifiable copy of the subscriber list */
152  subscrList = QF_PTR_AT_(QF_subscrList_, e->sig);
153  QF_CRIT_X_();
154 
155  if (QPSet_notEmpty(&subscrList)) { /* any subscribers? */
156  uint_fast8_t p;
157  QF_SCHED_STAT_
158 
159  QPSet_findMax(&subscrList, p); /* the highest-prio subscriber */
160 
161  QF_SCHED_LOCK_(p); /* lock the scheduler up to prio 'p' */
162  do { /* loop over all subscribers */
163  /* the prio of the AO must be registered with the framework */
164  Q_ASSERT_ID(210, QF_active_[p] != (QActive *)0);
165 
166  /* QACTIVE_POST() asserts internally if the queue overflows */
167  QACTIVE_POST(QF_active_[p], e, sender);
168 
169  QPSet_remove(&subscrList, p); /* remove the handled subscriber */
170  if (QPSet_notEmpty(&subscrList)) { /* still more subscribers? */
171  QPSet_findMax(&subscrList, p); /* highest-prio subscriber */
172  }
173  else {
174  p = 0U; /* no more subscribers */
175  }
176  } while (p != 0U);
177  QF_SCHED_UNLOCK_(); /* unlock the scheduler */
178  }
179 
180  /* The following garbage collection step decrements the reference counter
181  * and recycles the event if the counter drops to zero. This covers both
182  * cases when the event was published with or without any subscribers.
183  */
184  QF_gc(e);
185 }

206 void QActive_subscribe(QActive const * const me, enum_t const sig) {
207  uint_fast8_t p = (uint_fast8_t)me->prio;
208  QF_CRIT_STAT_
209 
210  Q_REQUIRE_ID(300, ((enum_t)Q_USER_SIG <= sig)
211  && (sig < QF_maxPubSignal_)
212  && (0U < p) && (p <= QF_MAX_ACTIVE)
213  && (QF_active_[p] == me));
214 
215  QF_CRIT_E_();
216 
217  QS_BEGIN_NOCRIT_PRE_(QS_QF_ACTIVE_SUBSCRIBE, me->prio)
218  QS_TIME_PRE_(); /* timestamp */
219  QS_SIG_PRE_(sig); /* the signal of this event */
220  QS_OBJ_PRE_(me); /* this active object */
221  QS_END_NOCRIT_PRE_()
222 
223  /* set the priority bit */
224  QPSet_insert(&QF_PTR_AT_(QF_subscrList_, sig), p);
225 
226  QF_CRIT_X_();
227 }

303 void QActive_unsubscribeAll(QActive const * const me) {
304  uint_fast8_t p = (uint_fast8_t)me->prio;
305  enum_t sig;
306 
307  Q_REQUIRE_ID(500, (0U < p) && (p <= QF_MAX_ACTIVE)
308  && (QF_active_[p] == me));
309 
310  for (sig = (enum_t)Q_USER_SIG; sig < QF_maxPubSignal_; ++sig) {
311  QF_CRIT_STAT_
312  QF_CRIT_E_();
313  if (QPSet_hasElement(&QF_PTR_AT_(QF_subscrList_, sig), p)) {
314  QPSet_remove(&QF_PTR_AT_(QF_subscrList_, sig), p);
315 
316  QS_BEGIN_NOCRIT_PRE_(QS_QF_ACTIVE_UNSUBSCRIBE, me->prio)
317  QS_TIME_PRE_(); /* timestamp */
318  QS_SIG_PRE_(sig); /* the signal of this event */
319  QS_OBJ_PRE_(me); /* this active object */
320  QS_END_NOCRIT_PRE_()
321  }
322  QF_CRIT_X_();
323 
324  /* prevent merging critical sections */
325  QF_CRIT_EXIT_NOP();
326  }
