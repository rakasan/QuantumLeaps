441 static int_fast8_t QHsm_tran_(QHsm * const me,
442  QStateHandler path[QHSM_MAX_NEST_DEPTH_],
443  uint_fast8_t const qs_id)
444 #else
445 static int_fast8_t QHsm_tran_(QHsm * const me,
446  QStateHandler path[QHSM_MAX_NEST_DEPTH_])
447 #endif
448 {
449  int_fast8_t ip = -1; /* transition entry path index */
450  int_fast8_t iq; /* helper transition entry path index */
451  QStateHandler t = path[0];
452  QStateHandler const s = path[2];
453  QState r;
454  QS_CRIT_STAT_
455 
456  /* (a) check source==target (transition to self)... */
457  if (s == t) {
458  QEP_EXIT_(s, qs_id); /* exit the source */
459  ip = 0; /* enter the target */
460  }
461  else {
462  (void)QEP_TRIG_(t, QEP_EMPTY_SIG_); /* find superstate of target */
463 
464  t = me->temp.fun;
465 
466  /* (b) check source==target->super... */
467  if (s == t) {
468  ip = 0; /* enter the target */
469  }
470  else {
471  (void)QEP_TRIG_(s, QEP_EMPTY_SIG_); /* find superstate of src */
472 
473  /* (c) check source->super==target->super... */
474  if (me->temp.fun == t) {
475  QEP_EXIT_(s, qs_id); /* exit the source */
476  ip = 0; /* enter the target */
477  }
478  else {
479  /* (d) check source->super==target... */
480  if (me->temp.fun == path[0]) {
481  QEP_EXIT_(s, qs_id); /* exit the source */
482  }
483  else {
484  /* (e) check rest of source==target->super->super..
485  * and store the entry path along the way
486  */
487  iq = 0; /* indicate that LCA not found */
488  ip = 1; /* enter target and its superstate */
489  path[1] = t; /* save the superstate of target */
490  t = me->temp.fun; /* save source->super */
491 
492  /* find target->super->super... */
493  r = QEP_TRIG_(path[1], QEP_EMPTY_SIG_);
494  while (r == (QState)Q_RET_SUPER) {
495  ++ip;
496  path[ip] = me->temp.fun; /* store the entry path */
497  if (me->temp.fun == s) { /* is it the source? */
498  iq = 1; /* indicate that LCA found */
499 
500  /* entry path must not overflow */
501  Q_ASSERT_ID(510,
502  ip < QHSM_MAX_NEST_DEPTH_);
503  --ip; /* do not enter the source */
504  r = (QState)Q_RET_HANDLED; /* terminate loop */
505  }
506  /* it is not the source, keep going up */
507  else {
508  r = QEP_TRIG_(me->temp.fun, QEP_EMPTY_SIG_);
509  }
510  }
511 
512  /* the LCA not found yet? */
513  if (iq == 0) {
514 
515  /* entry path must not overflow */
516  Q_ASSERT_ID(520, ip < QHSM_MAX_NEST_DEPTH_);
517 
518  QEP_EXIT_(s, qs_id); /* exit the source */
519 
520  /* (f) check the rest of source->super
521  * == target->super->super...
522  */
523  iq = ip;
524  r = (QState)Q_RET_IGNORED; /* LCA NOT found */
525  do {
526  if (t == path[iq]) { /* is this the LCA? */
527  r = (QState)Q_RET_HANDLED; /* LCA found */
528  ip = iq - 1; /* do not enter LCA */
529  iq = -1; /* cause termintion of the loop */
530  }
531  else {
532  --iq; /* try lower superstate of target */
533  }
534  } while (iq >= 0);
535 
536  /* LCA not found? */
537  if (r != (QState)Q_RET_HANDLED) {
538  /* (g) check each source->super->...
539  * for each target->super...
540  */
541  r = (QState)Q_RET_IGNORED; /* keep looping */
542  do {
543  /* exit t unhandled? */
544  if (QEP_TRIG_(t, Q_EXIT_SIG)
545  == (QState)Q_RET_HANDLED)
546  {
547  QS_BEGIN_PRE_(QS_QEP_STATE_EXIT, qs_id)
548  QS_OBJ_PRE_(me);
549  QS_FUN_PRE_(t);
550  QS_END_PRE_()
551 
552  (void)QEP_TRIG_(t, QEP_EMPTY_SIG_);
553  }
554  t = me->temp.fun; /* set to super of t */
555  iq = ip;
556  do {
557  /* is this LCA? */
558  if (t == path[iq]) {
559  /* do not enter LCA */
560  ip = (int_fast8_t)(iq - 1);
561  iq = -1; /* break out of inner loop */
562  /* break out of outer loop */
563  r = (QState)Q_RET_HANDLED;
564  }
565  else {
566  --iq;
567  }
568  } while (iq >= 0);
569  } while (r != (QState)Q_RET_HANDLED);

572  }
573  }
574  }
