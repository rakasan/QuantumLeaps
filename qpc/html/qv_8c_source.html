117 int_t QF_run(void) {
118 #ifdef Q_SPY
119  uint_fast8_t pprev = 0U; /* previously used priority */
120 #endif
121 
122  QF_onStartup(); /* application-specific startup callback */
123 
124  /* the combined event-loop and background-loop of the QV kernel... */
125  QF_INT_DISABLE();
126 
127  /* produce the QS_QF_RUN trace record */
128  QS_BEGIN_NOCRIT_PRE_(QS_QF_RUN, 0U)
129  QS_END_NOCRIT_PRE_()
130 
131  for (;;) {
132  QEvt const *e;
133  QActive *a;
134  uint_fast8_t p;
135 
136  /* find the maximum priority AO ready to run */
137  if (QPSet_notEmpty(&QV_readySet_)) {
138  QPSet_findMax(&QV_readySet_, p);
139  a = QF_active_[p];
140 
141 #ifdef Q_SPY
142  QS_BEGIN_NOCRIT_PRE_(QS_SCHED_NEXT, a->prio)
143  QS_TIME_PRE_(); /* timestamp */
144  QS_2U8_PRE_(p, /* priority of the scheduled AO */
145  pprev); /* previous priority */
146  QS_END_NOCRIT_PRE_()
147 
148  pprev = p; /* update previous priority */
149 #endif /* Q_SPY */
150 
151  QF_INT_ENABLE();
152 
153  /* perform the run-to-completion (RTC) step...
154  * 1. retrieve the event from the AO's event queue, which by this
155  * time must be non-empty and The "Vanialla" kernel asserts it.
156  * 2. dispatch the event to the AO's state machine.
157  * 3. determine if event is garbage and collect it if so
158  */
159  e = QActive_get_(a);
160  QHSM_DISPATCH(&a->super, e, a->prio);
161  QF_gc(e);
162 
163  QF_INT_DISABLE();
164 
165  if (a->eQueue.frontEvt == (QEvt *)0) { /* empty queue? */
166  QPSet_remove(&QV_readySet_, p);
167  }
168  }
169  else { /* no AO ready to run --> idle */
170 #ifdef Q_SPY
171  if (pprev != 0U) {
172  QS_BEGIN_NOCRIT_PRE_(QS_SCHED_IDLE, 0U)
173  QS_TIME_PRE_(); /* timestamp */
174  QS_U8_PRE_(pprev); /* previous priority */
175  QS_END_NOCRIT_PRE_()
176 
177  pprev = 0U; /* update previous priority */
178  }
179 #endif /* Q_SPY */
180 
181  /* QV_onIdle() must be called with interrupts DISABLED because
182  * the determination of the idle condition (no events in the
183  * queues) can change at any time by an interrupt posting events
184  * to a queue. QV_onIdle() MUST enable interrupts internally,
185  * perhaps at the same time as putting the CPU into a power-saving
186  * mode.
187  */
188  QV_onIdle();
189 
190  QF_INT_DISABLE();
191  }
192  }
193 #ifdef __GNUC__ /* GNU compiler? */
194  return 0;
195 #endif
196 }
